### 1.1.1 生成规则

```rust
impl PhysicalAppearance {
    /// 出生时生成——双亲混合 + 变异
    pub fn at_birth(
        mother: &PhysicalAppearance,
        father: &PhysicalAppearance,
        race: &RaceTraits,
        rng: &mut impl Rng,
    ) -> Self {
        PhysicalAppearance {
            height: blend(mother.height, father.height, 0.5, rng).clamp(-1.0, 1.0),
            build: if rng.gen_bool(0.15) { random_build(rng) }
                   else { *[&mother.build, &father.build].choose(rng).unwrap() },
            face_params: FaceParams::blend(&mother.face_params, &father.face_params, rng),
            skin_tone: SkinTone::blend(&mother.skin_tone, &father.skin_tone, rng),
            hair_color: HairColor::blend(&mother.hair_color, &father.hair_color, rng),
            grooming_level: 0.5,
            body_proportions: blend(mother.body_proportions, father.body_proportions, 0.5, rng),
            voice_pitch: blend(mother.voice_pitch, father.voice_pitch, 0.5, rng),
            voice_timbre: blend(mother.voice_timbre, father.voice_timbre, 0.5, rng),
            hair_style: HairStyle::default(),
            facial_hair: FacialHairStyle::default(),
            scars: vec![],
            tattoos: vec![],
            distinguishing_marks: vec![],
        }
    }

    /// 年龄影响
    pub fn apply_age(&mut self, age_pct: f32) {
        if age_pct > 0.5 { self.skin_tone.age(age_pct); }
        if age_pct > 0.6 { self.hair_color.gray_shift((age_pct - 0.6) / 0.4); }
        if age_pct > 0.85 { self.height -= 0.05; }
    }

    /// 从功能属性推导体型——连接两层
    pub fn derive_from_physical(phys: &PhysicalAttributes) -> BuildType {
        if phys.strength > 0.7 && phys.agility < 0.4 { return BuildType::Muscular; }
        if phys.strength > 0.6 && phys.agility > 0.5 { return BuildType::Lean; }
        if phys.strength < 0.3 && phys.agility < 0.3 { return BuildType::Heavy; }
        if phys.strength > 0.5 && phys.agility < 0.5 { return BuildType::Stocky; }
        if phys.agility > 0.7 { return BuildType::Slender; }
        BuildType::Average
    }
}
```

# Part 2: 文化审美标准

## 2.1 CulturalBeautyStandard

```rust
/// 每个文化有一个审美标准。存储在 CulturalSystem.values 中。
#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct CulturalBeautyStandard {
    pub ideal_build: BuildType,
    pub height_preference: f32,        // -1(偏好娇小) ~ 1(偏好高大)
    pub skin_tone_ideal: Option<SkinToneRange>,
    pub grooming_importance: f32,      // 0-1
    pub scar_stance: ScarStance,
    pub ideal_age_range: (f32, f32),   // 相对于成年年龄的比例
    pub appearance_weight_in_mating: f32,
    pub aesthetic_confidence: f32,
    pub formation_narrative: String,   // 仅叙事追溯
}

#[derive(Debug, Clone, Serialize, Deserialize)]
pub enum ScarStance { Honor, Neutral, Stigma }

impl CulturalBeautyStandard {
    /// 某人有多符合该文化的审美标准（0-1）
    pub fn conventional_attractiveness(&self, appearance: &PhysicalAppearance) -> f32 {
        let mut score = 0.0;
        let mut weight = 0.0;

        score += if appearance.build == self.ideal_build { 0.3 } else { 0.1 };
        weight += 0.3;

        let height_match = 1.0 - (appearance.height - self.height_preference).abs() / 2.0;
        score += height_match * 0.2;
        weight += 0.2;

        if let Some(ref ideal) = self.skin_tone_ideal {
            score += ideal.match_score(&appearance.skin_tone) * 0.15;
            weight += 0.15;
        }

        score += appearance.grooming_level * self.grooming_importance;
        weight += self.grooming_importance;

        match self.scar_stance {
            ScarStance::Honor => {
                score += (appearance.scars.len() as f32 * 0.05).min(0.15);
                weight += 0.15;
            }
            ScarStance::Stigma => {
                score -= (appearance.scars.len() as f32 * 0.08).min(0.15);
                weight += 0.15;
            }
            ScarStance::Neutral => {}
        }

        if weight == 0.0 { return 0.5; }
        (score / weight).clamp(0.0, 1.0)
    }
}

// SkinToneRange::match_score(): 实现阶段定义——色相+明度范围匹配，返回 0-1
```

### 2.1.1 形成与演化

审美标准来自三个来源：统治阶层外貌（精英模仿）、环境适应（热带偏好深肤色）、历史事件（战后伤疤=荣耀）。演化极慢——每代 ~0.05 偏移，完整转变需 10-20 代。跨文化接触加速演化。

```rust
/// 存储在 NpcData.attraction_template 中
#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct AttractionTemplate {
    // ── 取向 (先天为主) ──
    pub gender_attraction: BTreeMap<BiologicalSex, f32>,
    pub orientation_stability: f32,       // 0(终生可流动) ~ 1(终生固定)
    pub romantic_attraction: Option<BTreeMap<BiologicalSex, f32>>,

    // ── 去魅谱系 ──
    pub emotional_bond_requirement: f32,  // 0(即刻吸引) ~ 1(仅深厚情感后)
    pub attraction_separation: f32,      // 0(性与浪漫绑定) ~ 1(完全独立)

    // ── 偏好向量 (经验形成) ──
    pub personality_preference: PersonalityPreference,
    pub physical_preference: PhysicalPreferenceDeviation,
    pub status_preference: f32,           // -1(反感高地位) ~ 1(偏爱)
    pub skill_preference: BTreeMap<SkillCategory, f32>,

    // ── 关系偏好 ──
    pub relationship_preference: RelationshipPreference,
}

#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct PersonalityPreference {
    pub openness_pref: f32,           // -1(偏好低分) ~ 1(偏好高分)
    pub conscientiousness_pref: f32,
    pub extraversion_pref: f32,
    pub agreeableness_pref: f32,
    pub neuroticism_pref: f32,
    pub confidence: f32,              // 0(纯随机种子) ~ 1(经验验证)
}

#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct PhysicalPreferenceDeviation {
    pub build_deviation: f32,         // 偏离文化审美的个人偏好
    pub height_deviation: f32,
    pub face_deviation: [f32; 8],
    pub skin_tone_deviation: f32,
    pub voice_deviation: f32,
    pub grooming_deviation: f32,
}

impl PhysicalPreferenceDeviation {
    pub fn neutral() -> Self {
        Self {
            build_deviation: 0.0, height_deviation: 0.0,
            face_deviation: [0.0; 8], skin_tone_deviation: 0.0,
            voice_deviation: 0.0, grooming_deviation: 0.0,
        }
    }

    /// 个人偏好与目标外貌的匹配度 0-1
    pub fn match_score(&self, appearance: &PhysicalAppearance) -> f32 {
        let build_match = 1.0 - self.build_deviation.abs().min(1.0);
        let height_match = 1.0 - self.height_deviation.abs().min(1.0);
        let face_match = 1.0 - self.face_deviation.iter()
            .map(|d| d.abs()).sum::<f32>() / 8.0;
        let groom_match = 1.0 - self.grooming_deviation.abs().min(1.0);
        (build_match * 0.25 + height_match * 0.15 + face_match * 0.35 + groom_match * 0.15
         + 0.5 * 0.10).clamp(0.0, 1.0)  // voice + skin baseline
    }

    /// 所有偏差维度的绝对值平均——衡量"外貌主义"程度
    pub fn abs_average(&self) -> f32 {
        let face_avg = self.face_deviation.iter().map(|d| d.abs()).sum::<f32>() / 8.0;
        (self.build_deviation.abs() + self.height_deviation.abs() + face_avg
         + self.skin_tone_deviation.abs() + self.voice_deviation.abs()
         + self.grooming_deviation.abs()) / 6.0
    }
}
```

```rust
    }

    /// 从经历学习——每段重要关系塑造人格偏好
    pub fn learn_from_relationship(&mut self, partner_personality: &BigFive, affection_peak: f32) {
        let pref = &mut self.personality_preference;
        let learn_rate = 0.4 / (pref.confidence + 0.5);
        pref.openness_pref += (partner_personality.openness - 0.5 - pref.openness_pref) * learn_rate;
        pref.conscientiousness_pref += (partner_personality.conscientiousness - 0.5 - pref.conscientiousness_pref) * learn_rate;
        pref.extraversion_pref += (partner_personality.extraversion - 0.5 - pref.extraversion_pref) * learn_rate;
        pref.agreeableness_pref += (partner_personality.agreeableness - 0.5 - pref.agreeableness_pref) * learn_rate;
        pref.neuroticism_pref += (partner_personality.neuroticism - 0.5 - pref.neuroticism_pref) * learn_rate;
        pref.confidence = (pref.confidence + affection_peak * 0.15).min(1.0);
```

```rust
fn orientation_fluidity_check(
    template: &mut AttractionTemplate,
    relationship: &Relationship,
    target_sex: BiologicalSex,
) {
    if template.orientation_stability > 0.6 { return; }
    if relationship.affection > 0.7
        && relationship.trust > 0.7
        && relationship.familiarity > 0.6
        && template.gender_attraction.get(&target_sex).copied().unwrap_or(0.0) < 0.3

        let fluidity = 1.0 - template.orientation_stability;

    }
```

### 4.2.1 初始吸引力（第一次见面）

```rust
pub fn initial_attraction(
    observer: &NpcData,
    target: &NpcData,
    observer_template: &AttractionTemplate,
    beauty_standard: &CulturalBeautyStandard,
) -> f32 {
    // 0. 性别过滤
    let gender_match = observer_template.gender_attraction
        .get(&target.identity.biological_sex)
        .copied()
        .unwrap_or(0.1);
    if gender_match < 0.05 { return 0.0; }

    let app = &target.appearance;

    // 1. 外貌匹配 (权重 0.35)
    let culture_attractiveness = beauty_standard.conventional_attractiveness(app);
    let personal_deviation = observer_template.physical_preference.match_score(app);
    let physical_score = culture_attractiveness * 0.6 + personal_deviation * 0.4;

    // 2. 人格第一印象 (权重 0.15)
    let first_impression = target.personality.extraversion * 0.08
        + target.personality.openness * 0.05
```

```rust
    current: f32,
    interaction: &SocialInteraction,
    template: &AttractionTemplate,
    observer: &NpcData,
    target: &NpcData,
    relationship: &Relationship,

    match current {
        // 阶段一：陌生期 (≤0.25) —— 仅"有意义"的互动才增长
        a if a <= 0.25 => {
            if interaction.significance < 0.4 { return 0.0; }
            raw_delta * 0.3 * interaction.significance
        }
        // 阶段二：意识期 (0.25~0.55) —— 线性累积
        a if a <= 0.55 => {

                0.3 + relationship.familiarity * 0.7
            } else { 1.0 };
            raw_delta * 0.6 * demi_mod
        }
        // 阶段三：追求期 (0.55~0.78) —— 双向正反馈加速
        a if a <= 0.78 => {
            let mut growth = raw_delta;
            let self_reinforce = 0.3 + (a - 0.55) / (0.78 - 0.55) * 0.7;
            growth *= self_reinforce;
            // 感知互惠——来自行为观察，不是数据读取

            if perceived_recip > 0.5 {
                growth *= 1.0 + perceived_recip * 0.8;
            }
            if interaction.is_courtship() {
                growth *= 1.0 + interaction.effort_level * 0.4;
            }
            growth
        }
        // 阶段四：绑定 (≥0.78) —— 维持为主
        _ => {
            let maintenance = raw_delta * 0.1;
            let days_since = relationship.last_interaction.days_since();
            let decay = if days_since > 15.0 {
                -0.005 * (days_since - 15.0).sqrt()
            } else { 0.0 };
```

```rust
pub fn context_density_modifier(

    npc_b: &NpcData,
    context: &SocialContext,

    // 同聚落 → 高相遇频率
    if npc_a.identity.home_location() == npc_b.identity.home_location() {
        modifier *= 1.5;
    }

    // 同工作场所 → 每天必然见面
    if npc_a.identity.profession == npc_b.identity.profession {
        modifier *= 2.0;
    }

    // 共享创伤——一起活下来的人，一个月顶三年
    if npc_a.relationships.get(&npc_b.identity.id)
        .map(|r| r.source == RelationshipSource::SharedTrauma)
        .unwrap_or(false)
    {
        modifier *= 3.0;
    }

    // 航海/远征——24小时同处狭窄空间
    if let (Some(ref va), Some(ref vb)) = (&npc_a.vehicle_state, &npc_b.vehicle_state) {
        if va.vehicle_id == vb.vehicle_id {
            modifier *= 4.0;
        }
    }

    // 异地——一年见不了几次
    if npc_a.identity.home_location() != npc_b.identity.home_location()
        && context.distance_between(npc_a, npc_b) > 50.0
    {
        modifier *= 0.1;
    }

    modifier
}
```

```rust
/// 第三方观察: C 看 A 是否对 B 有意思
pub fn third_party_attraction_perception(
    observer: &NpcData,     // C
    subject: &NpcData,      // A
    target: &NpcData,       // B
    context: &SocialContext,
) -> Option<PerceivedAttraction> {
    // 观察者在场?
    if !observer.sensory_range_includes(subject) { return None; }

    let mut signals = 0.0;

    // 信号 1: 接近频率异常
    let proximity_bias = subject.proximity_bias_to(target, context);
    if proximity_bias > 0.3 { signals += 0.25; }

    // 信号 2: 公开求爱行为
    let public_courtship = subject.recent_public_courtship_toward(target);
    if public_courtship > 0 { signals += 0.35; }

    // 信号 3: 视线异常
    let gaze_bias = subject.gaze_bias_toward(target);
    if gaze_bias > 0.2 { signals += 0.20; }

    // 信号 4: 情绪变化
    let emotion_shift = subject.emotion_shift_when_near(target);
    if emotion_shift > 0.1 { signals += 0.20; }

    if signals < 0.3 { return None; }

    // 观察者能力: 物理感官 + 社交智慧
    let accuracy = observer.physique.perception * 0.6 + observer.mental.wisdom * 0.4;
    let confidence = signals * accuracy;

    if confidence > 0.4 {
        Some(PerceivedAttraction {
            subject_id: subject.identity.id,
            target_id: target.identity.id,
            perceived_intensity: confidence,
            observer_reaction: determine_observer_reaction(observer, subject, target),
        })
    } else {
        None
    }
}

        return ObserverReaction::Jealousy;  // 我自己也喜欢这个人
    }
    if observer.affection_to(subject.identity.id) > 0.6 {
        return ObserverReaction::SupportiveInterest;  // 好友→关心/撮合
    }
    // 从 CulturalSystem.norms 查询是否禁忌
    if cultural_system.is_forbidden_relationship(subject, target) {
        return ObserverReaction::Disapproval;
    }
    ObserverReaction::Gossip {
        spread_probability: 0.3 + observer.personality.extraversion * 0.3,
    }
}
```

```rust
    pre_breakup_attraction: f32,
    reason: &BreakupReason,
    personality: &BigFive,
    willpower: f32,
    days_since: f32,
) -> f32 {

        BreakupReason::TrustBroken => 0.05,
        BreakupReason::ExternalPressure => 0.01,
        BreakupReason::DeathOfPartner => 0.005,

        BreakupReason::NewAttraction => 0.04,

        BreakupReason::DeathOfPartner => 0.3,
```
