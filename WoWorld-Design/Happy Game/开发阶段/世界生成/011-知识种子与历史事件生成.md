### 2.2 传播算法

```rust
fn propagate_knowledge(
    seeds: &KnowledgeSeedRegistry,
    settlements: &[SettlementData],
    transport: &TransportNetwork,
    culture_params: &HashMap<SettlementId, CultureCoreParams>,
    years_to_simulate: u32,
    rng: &mut DeterministicRng,
) -> SettlementTechState {
    // 初始化: 每个聚落每项技艺 = 0 (未掌握)
    let mut tech_state = SettlementTechState::new(settlements);
    
    for year in (-years_to_simulate as i32)..=0_i32 {
        for era in seeds.eras_active_in(year) {
            for tech in &era.techs {
                // 1. 检查发现条件
                if tech.discovery_year == year {
                    for settlement in settlements {
                        if tech.condition.matches(settlement, year) {
                            // RootEvent: 此聚落首次发现该技术
                            tech_state.set_level(settlement.id, &tech.domain, tech.base_skill_ceiling);
                            tech_state.record_arrival(settlement.id, &tech.domain, year, None);
                        }
                    }
                }
                
                // 2. 传播: 对每对(已掌握A, 未掌握B)检查传播概率
                let knowers: Vec<SettlementId> = tech_state.settlements_with(&tech.domain);
                let non_knowers: Vec<SettlementId> = tech_state.settlements_without(&tech.domain);
                
                for &a_id in &knowers {
                    for &b_id in &non_knowers {
                        if !within_propagation_range(a_id, b_id, &tech, transport) { continue; }
                        
                        let prob = propagation_probability(
                            &tech, a_id, b_id, settlements, culture_params, transport,
                        );
                        
                        if rng.gen::<f32>() < prob {
                            // PropagationEvent: A→B 传播
                            let decayed_level = (tech_state.level(a_id, &tech.domain) as f32 * 0.7) as u8;
                            tech_state.set_level(b_id, &tech.domain, decayed_level);
                            tech_state.record_arrival(b_id, &tech.domain, year, Some(a_id));
                        }
                    }
                }
            }
        }
    }
    
    tech_state
}

fn propagation_probability(
    tech: &TechDef, from: SettlementId, to: SettlementId,
    settlements: &[SettlementData], culture: &HashMap<SettlementId, CultureCoreParams>,
    transport: &TransportNetwork,
) -> f32 {
    let distance = distance_between(from, to, settlements);
    let base_rate = match tech.propagation {
        PropagationSpeed::Fast => 0.30,
        PropagationSpeed::Slow => 0.05,
        PropagationSpeed::VerySlow => 0.01,
    };
    let trade_strength = transport.trade_connection_strength(from, to);
    let cultural_sim = culture_similarity(from, to, culture);
    let distance_decay = (-distance / tech.max_propagation_range()).exp();
    let receptivity = culture[&to].openness_to_change * (1.0 - culture[&to].traditionalism * 0.5);
    
    (base_rate * trade_strength * (0.5 + cultural_sim * 0.5) * distance_decay * receptivity)
        .clamp(0.0, 1.0)
}
```

**三类发现**：
- **Inevitable**: 条件仅需"有人类聚落+时间"→所有聚落在其时代内必然发现。
- **Conditional**: 需要特定资源/群系/前置技术→条件满足的聚落加权随机选择加权最高的 N 个。
- **Breakthrough**: 低概率随机事件 →满足前置+技术水平的聚落中以极低年概率触发。

## §三、因果事件图生成（P9b）

### 3.1 事件类型

| 事件类别 | 示例 | 根事件源 | 因果链路 |
|---------|------|---------|---------|
| 战争/冲突 | 领土战争·内战·宗教战争 | 资源竞争·权力继承·文化冲突 | 战争→人口减少→饥荒→移民 |
| 自然灾害 | 地震·火山·洪水·瘟疫 | 地理位置·群系·种子 | 灾害→人口锐减→建筑损毁 |
| 繁荣/发现 | 矿脉发现·贸易繁荣 | 资源分布·贸易路线 | 繁荣→人口增长→建筑升级 |
| 文化/宗教 | 宗教改革·文化复兴 | 文化参数·信仰分布 | 改革→信仰分化→冲突或融合 |
| 政治 | 王权更替·政体变革 | 权力拓扑·统治者死亡 | 更替→合法性波动→内战 |
| 探索 | 发现新大陆·跨洋接触 | 海洋系统·港口 | 接触→文化碰撞→贸易或征服 |

### 3.2 根事件采样

```rust
fn sample_root_events(
    settlements: &[SettlementData],
    npcs: &[NpcData],
    power: &PowerTopology,
    culture: &CultureCoreParams,
    years: u32,
    rng: &mut DeterministicRng,
) -> Vec<HistoricalEvent> {
    let mut events = Vec::new();
    let base_rate = ROOT_EVENT_PER_SETTLEMENT_PER_CENTURY;  // ~3-8
    
    for settlement in settlements {
        let n_events = poisson(base_rate * years as f32 / 100.0, rng);
        for _ in 0..n_events {
            let event_type = weighted_sample(&EVENT_TYPE_WEIGHTS, rng);
            let year = -1 * rng.gen_range(0..years) as i32;
            
            let event = match event_type {
                EventType::War => {
                    // 从邻近敌对政治实体或资源竞争者中选择对手
                    let opponent = select_war_opponent(settlement, power, rng);
                    HistoricalEvent::new_war(settlement, opponent, year, rng)
                }
                EventType::Plague => {
                    // 人口密度高 × 卫生条件差 → 概率↑
                    HistoricalEvent::new_plague(settlement, year, rng)
                }
                EventType::Discovery => {
                    // 从该聚落未掌握的稀有资源中随机
                    HistoricalEvent::new_discovery(settlement, year, rng)
                }
                // ...
            };
            events.push(event);
        }
    }
    
    events
}
```

### 3.3 因果链展开

```rust
fn expand_causal_chain(
    root_events: &[HistoricalEvent],
    world_state: &WorldState,
    years: u32,
    rng: &mut DeterministicRng,
) -> Vec<HistoricalEvent> {
    // 对每个根事件，以因果模板 × 当前世界状态 → 生成次级事件
    // 战争 → 饥荒(粮食产地被毁) → 移民 → 接收地人口压力 → 冲突
    
    let mut all_events = root_events.to_vec();
    let mut queue: VecDeque<_> = root_events.iter().collect();
    
    while let Some(event) = queue.pop_front() {
        for template in CAUSAL_TEMPLATES[&event.event_type] {
            if template.trigger_condition(event, world_state) {
                let child = template.generate(event, world_state, rng);
                all_events.push(child.clone());
                queue.push_back(&all_events.last().unwrap());
            }
        }
    }
    
    all_events
}
```
