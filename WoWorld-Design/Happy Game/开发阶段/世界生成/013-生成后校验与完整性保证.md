## §二、校验 1：引用完整性

所有 ID 交叉引用必须可解析。悬挂引用 = 崩溃。

```rust
fn validate_references(world: &WorldState) -> Result<(), Vec<ValidationError>> {
    let mut errors = Vec::new();
    
    // NPC → 建筑
    for npc in &world.npcs {
        if let Some(ref home) = npc.anchors.home {
            if let HomeAnchor::Building { building_id, .. } = home {
                if world.buildings.get(*building_id).is_none() {
                    errors.push(ValidationError::DanglingRef(
                        format!("NPC {:?} home building {:?} not found", npc.id, building_id)
                    ));
                }
            }
        }
        if let Some(ref wp) = npc.anchors.workplace {
            if let WorkplaceAnchor::Building { building_id, .. } = wp {
                if world.buildings.get(*building_id).is_none() {
                    errors.push(ValidationError::DanglingRef(
                        format!("NPC {:?} workplace building {:?} not found", npc.id, building_id)
                    ));
                }
            }
        }
        for owned in &npc.anchors.owned_buildings {
            if world.buildings.get(*owned).is_none() {
                errors.push(ValidationError::DanglingRef(
                    format!("NPC {:?} owned building {:?} not found", npc.id, owned)
                ));
            }
        }
    }
    
    // FamilyTree → NPC
    for edge in &world.family_tree.edges {
        if world.npcs.get(edge.from).is_none() {
            errors.push(ValidationError::DanglingFamilyRef(edge.from, edge.to, edge.relation));
        }
        if world.npcs.get(edge.to).is_none() {
            errors.push(ValidationError::DanglingFamilyRef(edge.from, edge.to, edge.relation));
        }
    }
    
    // PowerEdge → EntityId
    for edge in &world.power_topology.edges() {
        if !world.entity_exists(edge.holder) {
            errors.push(ValidationError::DanglingPowerHolder(edge.id, edge.holder));
        }
        if !world.entity_exists(edge.subject) {
            errors.push(ValidationError::DanglingPowerSubject(edge.id, edge.subject));
        }
    }
    
    // EventLog.participants → EntityId
    for event in &world.event_log {
        for participant in &event.participants {
            if !world.entity_exists(*participant) {
                errors.push(ValidationError::DanglingEventParticipant(event.id, *participant));
            }
        }
    }
    
    // Building.owner_npc_id → NpcId
    for building in &world.buildings {
        if let Some(owner) = building.owner_npc_id {
            if world.npcs.get(owner).is_none() {
                errors.push(ValidationError::DanglingBuildingOwner(building.id, owner));
            }
        }
    }
    
    if errors.is_empty() { Ok(()) } else { Err(errors) }
}
```

## §三、校验 2：时间一致性

```rust
fn validate_temporal_consistency(world: &WorldState) -> Result<(), Vec<ValidationError>> {
    let mut errors = Vec::new();
    let current_year: i32 = 0;  // 玩家进入时刻
    
    // 建筑年龄: built_year ≤ 0（任何建筑都在玩家进入前或等于建造）
    for b in &world.buildings {
        if b.built_year > current_year {
            errors.push(ValidationError::FutureBuilding(b.id, b.built_year));
        }
    }
    
    // NPC 出生: birth_year ≤ 0
    for npc in &world.npcs {
        if npc.birth_year > current_year {
            errors.push(ValidationError::FutureBirth(npc.id, npc.birth_year));
        }
    }
    
    // FamilyTree 父母出生 < 子女出生
    for edge in &world.family_tree.edges {
        match edge.relation {
            FamilyRelation::FatherOf | FamilyRelation::MotherOf => {
                let parent_age = world.npcs.get(edge.from).unwrap().birth_year;
                let child_age = world.npcs.get(edge.to).unwrap().birth_year;
                if parent_age >= child_age {
                    errors.push(ValidationError::TimeParadox(
                        format!("Parent {:?} born {} ≥ child {:?} born {}", 
                                edge.from, parent_age, edge.to, child_age)
                    ));
                }
            }
            _ => {}
        }
    }
    
    // 事件年份: ≤ 0 且 ≥ 聚落建成年代
    for event in &world.event_log {
        if event.timestamp > 0 {
            errors.push(ValidationError::FutureEvent(event.id, event.timestamp));
        }
    }
    
    if errors.is_empty() { Ok(()) } else { Err(errors) }
}
```

## §四、校验 3：人口一致性

```rust
fn validate_population(world: &WorldState) -> Result<(), Vec<ValidationError>> {
    let mut errors = Vec::new();
    
    // 每聚落实际 NPC 数 vs 宣称人口
    for settlement in &world.settlements {
        let actual_count = world.npcs.iter()
            .filter(|n| n.birth_settlement == settlement.id && n.death_year.is_none())
            .count();
        
        let declared_pop = settlement.population;
        
        // 允许 ±15% 偏差（死亡/迁徙未完全对齐）
        let tolerance = (declared_pop as f32 * 0.15) as usize;
        if (actual_count as i64 - declared_pop as i64).unsigned_abs() > tolerance as u64 {
            errors.push(ValidationError::PopulationMismatch(
                settlement.id, declared_pop, actual_count as u32
            ));
        }
    }
    
    // 总 NPC 数 ≈ Σ 聚落人口
    let total_declared: u32 = world.settlements.iter().map(|s| s.population).sum();
    let total_actual = world.npcs.iter().filter(|n| n.death_year.is_none()).count() as u32;
    if (total_declared as i64 - total_actual as i64).unsigned_abs() > total_declared as u64 / 10 {
        errors.push(ValidationError::TotalPopulationMismatch(total_declared, total_actual));
    }
    
    if errors.is_empty() { Ok(()) } else { Err(errors) }
}
```

## §五、校验 4：最小可玩性

```rust
fn validate_playability(world: &WorldState) -> Result<(), Vec<ValidationError>> {
    let mut errors = Vec::new();
    
    // 至少 1 个聚落 ≥ 50 人口
    if !world.settlements.iter().any(|s| s.population >= 50) {
        errors.push(ValidationError::NoViableSettlement);
    }
    
    // 至少 1 条可通行道路
    if world.transport.roads.edges.is_empty() {
        errors.push(ValidationError::NoRoads);
    }
    
    // 至少 1 个 Market 有 ≥ 10 个挂单
    if !world.economy.markets.iter().any(|m| m.total_listings() >= 10) {
        errors.push(ValidationError::NoViableMarket);
    }
    
    // 至少 1 个 NPC 有 ≥ 1 条已知价格
    if !world.npcs.iter().any(|n| !n.economic_state.known_prices.is_empty()) {
        errors.push(ValidationError::NoEconomicKnowledge);
    }
    
    if errors.is_empty() { Ok(()) } else { Err(errors) }
}
```

## §六、校验 5：性能预算核算

```rust
fn validate_budget(world: &WorldState) -> Result<(), Vec<ValidationError>> {
    let mut errors = Vec::new();
    let budget_rust_memory: usize = 1_400_000_000;  // 1.4GB
    
    let total_npc_memory = world.npcs.len() * 6000;  // ~6KB/NPC
    let total_power_memory = world.power_topology.edge_count() * 300;
    let total_event_memory = world.event_log.len() * 2000;
    let total_building_memory = world.buildings.len() * 1000;
    
    let estimated_total = total_npc_memory + total_power_memory 
        + total_event_memory + total_building_memory + 250_000_000;  // +250MB 其他
    
    if estimated_total > budget_rust_memory {
        errors.push(ValidationError::MemoryBudgetExceeded(estimated_total, budget_rust_memory));
    }
    
    if errors.is_empty() { Ok(()) } else { Err(errors) }
}
```

fn apply_world_delta(world: &mut WorldState, delta: &WorldStateDelta) {
    // P9 产出的历史副作用——标记为废墟/废弃/死亡地点
    for (building_id, condition) in &delta.buildings_damaged {
        if let Some(b) = world.buildings.get_mut(*building_id) {
            b.condition = *condition;
        }
    }
    for settlement_id in &delta.settlements_abandoned {
        if let Some(s) = world.settlements.get_mut(*settlement_id) {
            s.status = SettlementStatus::Abandoned;
        }
    }
    // 大规模死亡地点 → 亡灵生成锚点（已由 P3.5 亡灵部分消费）
}

fn finalize_graves(world: &mut WorldState, grave_specs: &[GraveSpec], rng: &mut DeterministicRng) {
    for settlement in &world.settlements {
        let graves: Vec<&GraveSpec> = grave_specs.iter()
            .filter(|g| g.settlement_id == settlement.id)
            .filter(|g| g.death_year > -100)  // 仅近 100 年
            .collect();
        
        if graves.is_empty() { continue; }
        
        let cemetery_zone = world.get_cemetery_zone(settlement.id);
        let grave_buildings = building_module::generate_graves(&graves, &cemetery_zone, rng);
        world.buildings.extend(grave_buildings);
    }
}

## §九、校验失败处理

如果校验失败：
- **引用/时间不一致** → 致命错误。加载画面显示"世界种子无效"，建议重试或换种子。
- **人口偏差** → 警告。自动调整（调整聚落宣称人口以匹配实际数）。
- **最小可玩性不满足** → 致命错误。"这是一个不适合生存的世界。请尝试另一个种子。"
- **内存超预算** → 警告。记录实际预算使用量用于后续优化。不阻止加载。
