```rust
use woworld_core::prelude::WorldPos;

/// 单层密度场——正值=实体，负值=空
pub trait DensityProvider: Send + Sync {
    fn density_at(&self, pos: WorldPos) -> f32;
    fn material_at(&self, pos: WorldPos) -> u8;
    fn priority(&self) -> u8;
    fn layer_name(&self) -> &'static str { "unnamed" }
}
```

### 2.2 `DensityStack`（Step 0 创建）

```rust
pub struct DensityStack {
    layers: Vec<Box<dyn DensityProvider>>, // 按 priority 升序
}

impl DensityStack {
    pub fn new() -> Self { Self { layers: Vec::new() } }
    pub fn push(&mut self, layer: Box<dyn DensityProvider>) {
        let pos = self.layers
            .binary_search_by_key(&layer.priority(), |l| l.priority())
            .unwrap_or_else(|e| e);
        self.layers.insert(pos, layer);
    }
    pub fn density_at(&self, pos: WorldPos) -> f32 {
        self.layers.iter().fold(0.0f32, |acc, l| acc + l.density_at(pos))
    }
}

#[cfg(test)]
mod tests {
```

```rust
use woworld_core::density::DensityProvider;
use std::sync::Arc;

pub struct TerrainBaseDensity {
    noise: Arc<WorldNoise>,

impl TerrainBaseDensity {
    pub fn new(noise: Arc<WorldNoise>) -> Self { Self { noise } }
}

impl DensityProvider for TerrainBaseDensity {
    fn density_at(&self, pos: WorldPos) -> f32 {
        let h = self.noise.sample_height(pos.x, pos.z) as f32;
        pos.y as f32 - h

    fn material_at(&self, pos: WorldPos) -> u8 {

}
```

```rust
/// 一次调用同时获取高度、法线、表面材质
/// ——零额外噪声开销（4 次 Perlin vs 独立调用需 7-10 次）
pub fn sample_vertex(&self, pos: WorldPos) -> (f32, Vec3, SurfaceMaterial) {
    let h = self.density_at(pos);
    let n = self.calc_normal(pos);      // 3 次 density_at (有限差分, eps=0.5)
    let m = self.surface_material_at(pos); // 复用 h 和 n + 1 次 density_at (斜率检查)
    (h, n, m)
}
```

**`generate_heightmap_data()` 精确签名**:
```rust
use std::collections::HashMap;
use woworld_core::material::SurfaceMaterial;

pub fn generate_heightmap_data(
    terrain: &HeightfieldTerrain,
    center_x: f64,
    center_z: f64,
    spacing: f64,
    texture_size: u32,
    material_colors: &HashMap<SurfaceMaterial, [f32; 4]>,
) -> HeightmapData {
    use woworld_core::spatial::TerrainQuery;
    let half = texture_size as f64 * spacing * 0.5;
    let ox = center_x - half;
    let oz = center_z - half;
    let pixel_count = (texture_size * texture_size) as usize;
    let mut heights = Vec::with_capacity(pixel_count);
    let mut material_colors_out = Vec::with_capacity(pixel_count);

    for iz in 0..texture_size {
        let wz = oz + iz as f64 * spacing;
        for ix in 0..texture_size {
            let wx = ox + ix as f64 * spacing;
            let pos = WorldPos { x: wx, y: 0.0, z: wz };
            let (h, _normal, material) = terrain.sample_vertex(pos);
            heights.push(h);
            let color = material_colors
                .get(&material)
                .copied()
                .unwrap_or([0.76, 0.70, 0.50, 1.0]); // fallback: sand
            material_colors_out.push(color);
        }
    }
    HeightmapData { heights, material_colors: material_colors_out }
}
```

> ⚠️ **性能**: 每像素 4 次 Perlin（vs 旧 `height_at()` 的 1-3 次），+33% CPU 开销。全部在 rayon 后台线程——主线程零感知。256²=65K 像素 ~2.6ms，512²=262K 像素 ~10.5ms（单线程；rayon 16 线程下各层并行，L4-L7 重叠执行）。

**terrain.gdshader 完整代码**:
```glsl
shader_type spatial;
render_mode cull_back;

uniform sampler2D heightmap : source_color, filter_linear, repeat_disable;
uniform sampler2D material_map : source_color, filter_linear, repeat_disable;
uniform sampler2D neighbor_heightmap : source_color, filter_linear, repeat_disable;
uniform vec3 camera_pos = vec3(0.0);
uniform vec3 node_pos = vec3(0.0);
uniform vec2 grid_origin = vec2(0.0);
uniform float hm_extent = 256.0;

void vertex() {
    vec3 world_pos = VERTEX + node_pos;

    if (VERTEX.y > -300.0) {
        // 地表顶点：VTF 高度采样
        vec2 uv = (world_pos.xz - grid_origin) / hm_extent;
        float h = texture(heightmap, uv).r;

        // LOD 过渡混合
        float blend = CUSTOM0.x;
        if (blend > 0.001) {
            float h_neighbor = texture(neighbor_heightmap, uv).r;
            h = mix(h, h_neighbor, blend);
        }
        world_pos.y = h;
    }
    // else: 裙边底点保持 y=-400

    // camera-relative view transform
    vec3 rel = world_pos - camera_pos;
    vec3 view_pos = mat3(VIEW_MATRIX) * rel;
    POSITION = PROJECTION_MATRIX * vec4(view_pos, 1.0);

    if (VERTEX.y > -300.0) {
        // 梯度法线
        float ts = hm_extent / float(textureSize(heightmap, 0).x);
        vec2 uv = (world_pos.xz - grid_origin) / hm_extent;
        float h_c = texture(heightmap, uv).r;
        float h_r = texture(heightmap, uv + vec2(ts / hm_extent, 0.0)).r;
        float h_u = texture(heightmap, uv + vec2(0.0, ts / hm_extent)).r;
        float gx = (h_r - h_c) / ts;
        float gz = (h_u - h_c) / ts;
        vec3 world_normal = normalize(vec3(-gx, 1.0, -gz));
        NORMAL = mat3(VIEW_MATRIX) * world_normal;
    } else {
        NORMAL = mat3(VIEW_MATRIX) * vec3(0.0, 1.0, 0.0);
    }

    // 材质纹理——替代硬编码高度色带
    COLOR = texture(material_map, (world_pos.xz - grid_origin) / hm_extent);
}

void fragment() {
    ALBEDO = COLOR.rgb;
}
```

**blend_factor 计算算法**（`generate_clipmap_grid()` 末尾）:
```rust
// blend_factor: 外边界 2×spacing 环内顶点从 0→1（向最外圈递增）
let mut blend_factors = vec![0.0f32; vertices.len()];
let transition_width = 2.0 * spacing;
for iz in 0..n_usize {
    for ix in 0..n_usize {
        let curr = vertex_map[iz * n_usize + ix];
        if curr.is_none() { continue; }
        let ci = curr.unwrap() as usize;
        let dist_from_center = (vertices[ci].x.powi(2) + vertices[ci].z.powi(2)).sqrt();
        let outer_ring = level.max_range - dist_from_center;
        if outer_ring <= transition_width && outer_ring >= 0.0 {
            blend_factors[ci] = (1.0 - outer_ring / transition_width) as f32;
        }
    }
}
```

> ⚠️ **Godot 4.7 关键坑位**:
> 1. `ARRAY_CUSTOM0` 对应 `Array::set(6, ...)`（非 7）——Godot 枚举：0=VERTEX, 1=NORMAL, 2=TANGENT, 3=COLOR, 4=TEX_UV, 5=TEX_UV2, **6=CUSTOM0**, 7=CUSTOM1, 8=CUSTOM2, 9=CUSTOM3, 12=INDEX
> 2. `NORMAL` 在 spatial vertex shader 中必须是**视空间**（view space）——已通过 `mat3(VIEW_MATRIX) * world_normal` 转换
> 3. `CUSTOM0` 是 `vec4`——单通道 `PackedFloat32Array` 写入 `CUSTOM0.x`，`.yzw` 默认为 0
> 4. `textureSize(sampler2D, 0)` 在 Godot 4.x 中有效——LOD 0（基础层）返回纹理分辨率
> 5. `ResourceLoader::load()` 失败不 panic——返回 `Option<Gd<Resource>>`，需 `.expect("描述性错误信息")`
> 6. `cull_back` 与 camera-relative 变换兼容——视空间顶点绕序保持 CCW（已验证现有网格生成绕序 `[tl, bl, tr, tr, bl, br]`）
> 7. 避免 Bug #59517（GPU buffer flush stutter）——用双缓冲 `ImageTexture` 创建替代 `update()`（Sprint 025 已验证）
