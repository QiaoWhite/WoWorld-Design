## 一、BuildingStylePreferences —— 文化→建筑风格的映射

### 1.1 数据结构

```rust
/// 文化建筑风格偏好 —— 第一层派生参数
/// 
/// 从 CultureCoreParams + 群系环境参数 派生。
/// 输入到世界生成 P6（建筑 WFC）决定建筑视觉效果。
/// 
/// 消费者：世界生成 005（建筑 WFC）、家居系统
#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct BuildingStylePreferences {
    /// 屋顶样式偏好
    pub roof: RoofStyle,

    /// 墙体材质偏好
    pub wall_material: WallMaterialPreference,

    /// 装饰复杂度
    pub decoration: DecorationLevel,

    /// 颜色调色板
    pub palette: ColorPalette,

    /// 建筑对称性偏好
    /// 0.0 = 非对称、有机布局（自然生长式）
    /// 1.0 = 严格对称（规整、几何）
    pub symmetry: f32,

    /// 建筑尺度倾向
    /// 0.0 = 亲密的、小尺度（建筑体量紧凑）
    /// 1.0 = 宏大的、纪念性的（高天花板/大空间/高大门）
    pub scale: f32,

    /// 房间邻接文化修正
    /// 不同文化对房间布局有不同的"应该挨着什么"的偏好
    pub adjacency_modifiers: AdjacencyModifiers,
}

/// 屋顶样式
#[derive(Debug, Clone, Copy, PartialEq, Eq, Serialize, Deserialize)]
pub enum RoofStyle {
    Flat,        // 平顶——干旱/沙漠文化。可用于户外生活空间
    Sloped,      // 坡顶——温带/多雨文化。排水优先
    Dome,        // 穹顶——宗教/东方文化。象征天穹
    Spire,       // 尖顶——哥特/北方文化。垂直→向上仰望
    Curved,      // 曲线——东方精致。柔和过渡
    Longhouse,   // 长屋——北欧/集体文化。一栋建筑容纳多人/多代
    Thatched,    // 茅草——贫困/草原。易建、价廉
}

/// 墙体材质偏好
#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct WallMaterialPreference {
    /// 首选材质
    pub primary: WallMaterial,
    /// 备选材质（如果首选材料在当地不可获取）
    pub fallback: WallMaterial,
    /// 装饰石材的比例（0-1）——即便是当地石材，发达文化可能更多用进口石材装饰
    pub imported_stone_ratio: f32,
}

#[derive(Debug, Clone, Copy, PartialEq, Eq, Serialize, Deserialize)]
pub enum WallMaterial {
    LocalStone,    // 当地石材——颜色从当地地质决定
    Timber,        // 木材——森林区
    Brick,         // 砖瓦——发达聚落、河流区（粘土来源）
    Adobe,         // 土坯——沙漠/贫困
    WattleDaub,    // 编条抹泥——中世纪平民（木材框架+粘土填充）
    ImportedStone, // 进口石材——极富裕（大理石、花岗岩远距运输）
    Ice,           // 冰雪——极地/亚寒带
    Hide,          // 兽皮——游牧/半游牧
}

/// 装饰复杂度
#[derive(Debug, Clone, Copy, PartialEq, Eq, Serialize, Deserialize)]
pub enum DecorationLevel {
    Bare,        // 几乎无——功能至上
    Minimal,     // 简单功能性装饰（窗框/门框基本雕刻）
    Moderate,    // 平衡的装饰（雕刻/绘画点缀）
    Ornate,      // 大量装饰（丰富雕刻、壁画、马赛克）
    Extravagant, // 极尽装饰——每一表面都有艺术处理
}

/// 颜色调色板
#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct ColorPalette {
    /// 主色调（墙壁/外观主色）
    pub primary_hue: HueFamily,
    /// 强调色（门窗/装饰线条）
    pub accent_hue: HueFamily,
    /// 色彩饱和度
    /// 0.0 = 大地色/灰色（低调、自然）
    /// 1.0 = 鲜艳/饱和（高调、引人注目）
    pub saturation: f32,
    /// 色彩对比度
    /// 0.0 = 单色系/弱对比（和谐统一）
    /// 1.0 = 强对比（活泼、戏剧性）
    pub contrast: f32,
}

#[derive(Debug, Clone, Copy, PartialEq, Eq, Serialize, Deserialize)]
pub enum HueFamily {
    Warm,      // 红/橙/黄系列
    Cool,      // 蓝/绿/紫系列
    Earth,     // 棕/米/灰系列
    Neutral,   // 白/灰/黑
    Vibrant,   // 随机鲜艳色——每个建筑可能不同
}

/// 房间邻接文化修正
/// 
/// 不同文化对"什么房间应该挨着什么房间"有不同的偏好。
/// 叠加在基础邻接规则之上（见 世界生成 005 §2.3）。
#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct AdjacencyModifiers {
    /// 厨房与主厅——需要相邻（共用炉火/供暖）
    /// 寒带文化=强制相邻，热带文化=无关紧要
    pub kitchen_mainhall_required: bool,
    /// 外部门廊/玄关——是否需要
    /// 高个人空间文化=需要缓冲区
    pub entry_buffer_preferred: bool,
    /// 庭院偏好——是否需要室内或半室内庭院
    pub courtyard_preference: f32,  // 0=不需要, 1=核心组织元素
    /// 功能分区严格度——不同功能的房间是否必须隔离
    /// 0=开放布局（多功能空间、开放式平面）
    /// 1=严格分区（每个功能独立房间）
    pub zoning_strictness: f32,
}
```

### 1.2 BuildingStylePreferences 的派生公式

```rust
impl BuildingStylePreferences {
    /// 从核心文化参数 + 群系环境 派生建筑风格偏好
    /// 
    /// 此为第一层派生——世界生成时计算一次并缓存。
    /// 重算触发器：核心参数显著变化 或 群系重分类（极少发生）。
    pub fn derive_from(core: &CultureCoreParams, biome: &BiomeParams) -> Self {
        let roof = derive_roof_style(core, biome);
        let wall = derive_wall_material(core, biome);
        let decoration = derive_decoration_level(core);
        let palette = derive_color_palette(core, biome);
        let symmetry = derive_symmetry(core);
        let scale = derive_scale(core);
        let adjacency = derive_adjacency_modifiers(core, biome);
        
        Self { roof, wall_material: wall, decoration, palette, symmetry, scale, adjacency_modifiers: adjacency }
    }
}

/// 屋顶样式 = f(群系降水, 艺术性, 权力距离, 个人主义, 不确定性规避)
fn derive_roof_style(core: &CultureCoreParams, biome: &BiomeParams) -> RoofStyle {
    // 首要决定因素：气候
    if biome.is_arid {
        return RoofStyle::Flat; // 干旱 → 平顶（不需要排水，可用作户外空间）
    }
    if biome.is_arctic || biome.is_tundra {
        return RoofStyle::Sloped; // 寒带 → 坡顶（排雪）+ 可能 Ice 材质
    }
    
    // 温带及湿润区域——由文化选择
    // 高艺术性 + 高宗教性 → 穹顶
    if core.artistry > 0.7 && core.religiosity > 0.6 {
        return RoofStyle::Dome;
    }
    // 高竞争 + 高权力距离 → 尖顶（向上追求——竞争精神的物质化）
    if core.competition_orientation > 0.7 && core.power_distance > 0.6 {
        return RoofStyle::Spire;
    }
    // 高集体 + 高长期取向 → 长屋（多代同堂/集体居住）
    if (1.0 - core.individualism) > 0.7 && core.long_term_orientation > 0.6 {
        return RoofStyle::Longhouse;
    }
    // 高艺术性 + 低竞争 → 曲线（柔和审美）
    if core.artistry > 0.65 && core.competition_orientation < 0.4 {
        return RoofStyle::Curved;
    }
    // 温带森林 → 茅草顶（易建、价廉、材料丰富）
    if biome.is_forested {
        return RoofStyle::Thatched;
    }
    // 低不确定性规避 → 坡顶（务实、标准的排水方案）
    // 高不确定性规避 → 坡顶（传统、稳固的样式——但不能是平的因为下雨）
    RoofStyle::Sloped
}

/// 墙体材质 = f(群系可用材料, 经济水平, 长期取向, 艺术性)
/// 
/// 注：实际材质选择还需结合当地可用材料——世界生成 P6 负责最终决定
fn derive_wall_material(core: &CultureCoreParams, biome: &BiomeParams) -> WallMaterialPreference {
    let primary = if biome.is_forested {
        WallMaterial::Timber
    } else if biome.is_desert {
        WallMaterial::Adobe
    } else if biome.is_arctic {
        WallMaterial::Ice
    } else {
        WallMaterial::LocalStone
    };
    
    let imported_ratio = if core.artistry > 0.7 && core.long_term_orientation > 0.5 {
        0.15 // 重视美感和质量 → 愿意进口好材料
    } else {
        0.03
    };
    
    WallMaterialPreference {
        primary,
        fallback: WallMaterial::WattleDaub, // 通用廉价备选
        imported_stone_ratio: imported_ratio,
    }
}

/// 装饰复杂度 = f(艺术性, 权力距离, 宗教性, 个人主义)
fn derive_decoration_level(core: &CultureCoreParams) -> DecorationLevel {
    let raw = core.artistry * 0.55           // 艺术重视度——最大驱动
            + core.power_distance * 0.15     // 等级→炫耀性装饰→"宫殿必须宏伟"
            + core.religiosity * 0.15        // 宗教建筑装饰→教堂/神殿装饰
            + core.individualism * 0.15;      // 个人主义→个性化装饰表达
    
    if raw > 0.8 { DecorationLevel::Extravagant }
    else if raw > 0.6 { DecorationLevel::Ornate }
    else if raw > 0.35 { DecorationLevel::Moderate }
    else if raw > 0.15 { DecorationLevel::Minimal }
    else { DecorationLevel::Bare }
}

/// 颜色调色板 = f(艺术性, 个性主义, 放纵度, 群系)
fn derive_color_palette(core: &CultureCoreParams, _biome: &BiomeParams) -> ColorPalette {
    // 主色调：从群系决定基础方向
    let primary = HueFamily::Earth; // 大多数文化使用大地色为主调
    
    // 强调色：从文化性格决定
    let accent = if core.artistry > 0.6 && core.indulgence > 0.5 {
        HueFamily::Vibrant // 热爱艺术+享受→鲜艳多样化
    } else if core.artistry > 0.6 {
        HueFamily::Warm // 艺术→暖色为主
    } else {
        HueFamily::Cool // 功能→冷色
    };
    
    // 饱和度 = f(艺术性, 放纵度, 权力距离)
    let saturation = (core.artistry * 0.5 + core.indulgence * 0.3 + (1.0 - core.power_distance) * 0.2).clamp(0.0, 1.0);
    
    // 对比度 = f(竞争, 艺术性)
    let contrast = (core.competition_orientation * 0.5 + core.artistry * 0.5).clamp(0.0, 1.0);
    
    ColorPalette { primary_hue: primary, accent_hue: accent, saturation, contrast }
}

/// 对称性 = f(不确定性规避, 权力距离, 艺术性)
/// 高不确定性规避 → 严格对称（规矩、秩序）
/// 高权力距离 → 对称（对称建筑=权力和永恒的象征）
/// 高艺术性 + 低不确定性规避 → 不对称（自由表达的有机美）
fn derive_symmetry(core: &CultureCoreParams) -> f32 {
    (core.uncertainty_avoidance * 0.45 + core.power_distance * 0.35 + (1.0 - core.artistry) * 0.20).clamp(0.0, 1.0)
}

/// 建筑尺度 = f(权力距离, 竞争取向, 集体主义)
/// 高权力距离 → 纪念性（国王/神的需要宏大——"大=权力"）
/// 高竞争取向 → 个人炫耀——攀比建筑规模
/// 高集体主义 → 大规模公共建筑，个人的内部空间小
fn derive_scale(core: &CultureCoreParams) -> f32 {
    (core.power_distance * 0.45 + core.competition_orientation * 0.30 + (1.0 - core.individualism) * 0.25).clamp(0.0, 1.0)
}

/// 房间邻接文化修正
fn derive_adjacency_modifiers(core: &CultureCoreParams, biome: &BiomeParams) -> AdjacencyModifiers {
    AdjacencyModifiers {
        // 寒冷文化 → 厨房必须挨着主厅（共用炉火取暖）
        kitchen_mainhall_required: biome.is_cold || core.long_term_orientation > 0.7,
        // 高个人空间 → 需要玄关作为"外部→内部"缓冲区
        entry_buffer_preferred: core.individualism > 0.6,
        // 庭院偏好：温暖气候 → 室外生活；高集体主义 → 共享庭院
        courtyard_preference: if biome.is_warm { 0.6 } else { 0.2 }
                            + (1.0 - core.individualism) * 0.4,
        // 分区严格度：高不确定性规避 → 每物各有其位
        zoning_strictness: core.uncertainty_avoidance * 0.7 + core.power_distance * 0.3,
    }
}
```

## 二、CulturalBeautyStandard —— 文化审美标准

### 2.1 数据结构

```rust
/// 文化审美标准 —— 一个文化中"什么是美"的集体共识
/// 
/// 第一层派生参数——从 CultureCoreParams + 统治阶层外观 + 环境适应 派生。
/// 
/// 当前定义在 NPC 02-性别与吸引力系统 §2.1 中。
/// 所有权随文化系统创建而迁至本文档——NPC 模块保留消费逻辑。
/// 
/// 消费方：NPC 吸引力计算、跨文化婚配行为
#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct CulturalBeautyStandard {
    /// 理想体型
    pub ideal_build: BuildType,

    /// 身高偏好：-1.0 = 偏好娇小, 0.0 = 无偏好, 1.0 = 偏好高大
    pub height_preference: f32,

    /// 理想肤色范围（可选——None 表示无肤色偏好）
    pub skin_tone_ideal: Option<SkinToneRange>,

    /// 修饰/装饰重要性——一个"好看"的人有多依赖于主动修饰
    /// 0.0 = 自然美（修饰不重要——天然的样子就是美）
    /// 1.0 = 修饰极重要（精致的服饰/化妆/发型决定吸引力）
    pub grooming_importance: f32,

    /// 伤疤的社交含义
    pub scar_stance: ScarStance,

    /// 理想年龄范围
    pub ideal_age_range: (f32, f32),

    /// 外表在婚配中的权重——选择伴侣时有外表因素多少
    /// 0.0 = 外表完全不重要
    /// 1.0 = 外表决定一切
    pub appearance_weight_in_mating: f32,

    /// 审美自信——该文化对自己的审美有多自信
    /// 0.0 = 易受外来审美影响（"他们的生活方式可能更好"）
    /// 1.0 = 审美封闭（"我们的审美是唯一正确的"）
    pub aesthetic_confidence: f32,
}

#[derive(Debug, Clone, Copy, PartialEq, Eq, Serialize, Deserialize)]
pub enum BuildType {
    Lean,       // 瘦/精瘦
    Athletic,   // 健壮/运动型
    Robust,     // 魁梧/敦实
    Plump,      // 丰满/富态
    Unspecified, // 无显著偏好
}

#[derive(Debug, Clone, Copy, PartialEq, Eq, Serialize, Deserialize)]
pub enum ScarStance {
    /// 伤疤 = 荣誉标记（武士文化——每一道伤疤讲一个故事）
    Honor,
    /// 伤疤 = 中性（无所谓）
    Neutral,
    /// 伤疤 = 缺陷（应被遮掩——特别是面部伤疤）
    Stigma,
}
```

### 2.2 CulturalBeautyStandard 的派生公式

```rust
impl CulturalBeautyStandard {
    /// 从核心文化参数 + 统治阶层/精英的外观特征 派生审美标准
    /// 
    /// 此为第一层派生——世界生成时计算。随核心参数缓慢漂移更新。
    /// 
    /// 进化速率：约 0.05/代际（~25年）。完全转变需 10-20 代。
    pub fn derive_from(core: &CultureCoreParams, elite_appearance: &EliteAppearance) -> Self {
        Self {
            ideal_build: derive_ideal_build(core),
            height_preference: derive_height_preference(core),
            skin_tone_ideal: derive_skin_tone_ideal(core, elite_appearance),
            grooming_importance: derive_grooming_importance(core),
            scar_stance: derive_scar_stance(core),
            ideal_age_range: derive_ideal_age_range(core),
            appearance_weight_in_mating: derive_appearance_weight(core),
            aesthetic_confidence: derive_aesthetic_confidence(core),
        }
    }
}

/// 理想体型 = f(军事化, 放纵度, 艺术性)
/// 高军事化 → Athletic（健壮=战斗力）
/// 高放纵度 + 低军事化 → Plump（丰满=享受生活的证据→富贵象征）
/// 高艺术性 + 高个体 → Lean（艺术家的敏感体型）
fn derive_ideal_build(core: &CultureCoreParams) -> BuildType {
    if core.militarism > 0.6 { BuildType::Athletic }
    else if core.indulgence > 0.6 && core.militarism < 0.4 { BuildType::Plump }
    else if core.artistry > 0.6 && core.individualism > 0.5 { BuildType::Lean }
    else if core.power_distance > 0.7 { BuildType::Robust } // 等级社会→魁梧=权威
    else { BuildType::Unspecified }
}

/// 伤疤社会态度 = f(军事化, 竞争取向, 权力距离)
/// 高军事 → 伤疤=荣誉（战士勋章）
/// 高竞争 + 低军事 → 伤疤=中性或污点（完美外表=竞争优势）
fn derive_scar_stance(core: &CultureCoreParams) -> ScarStance {
    if core.militarism > 0.5 { ScarStance::Honor }
    else if core.competition_orientation > 0.6 { ScarStance::Neutral }
    else { ScarStance::Neutral } // 大多数文化=中性
}
```

**注**：关于精英外观（`elite_appearance`）的输入——统治阶层的外貌特征提供审美标准的"模仿基准"。NPC 倾向于模仿社会上层的外观。此机制在 NPC 模块 02-性别与吸引力系统 中完整定义，文化系统不重新定义——仅通过 `derive_from` 将此参数纳入公式。

## 三、DietaryBasePreferences —— 饮食文化基础

### 3.1 数据结构

```rust
/// 文化的饮食基础偏好 —— 第一层派生参数
/// 
/// 从核心参数 + 群系可用食材 派生。
/// 输入到经济系统的消费偏好推导。
/// 
/// 消费者：经济系统（需求结构修正）、NPC（食物选择行为）
#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct DietaryBasePreferences {
    /// 主食类型
    pub staple: StapleType,

    /// 肉食文化 —— 肉类在饮食中的角色
    pub meat_role: MeatRole,

    /// 饮酒文化 —— 对酒精的接受度
    pub alcohol_stance: AlcoholStance,

    /// 共食 vs 分食 —— 一起分享还是每人独立一份
    pub dining_style: DiningStyle,

    /// 香料使用倾向
    /// 0.0 = 极少香料（口味清淡）
    /// 1.0 = 大量香料（口味丰富——通常是热带文化，因为香料植物多）
    pub spice_preference: f32,
}

#[derive(Debug, Clone, Copy, PartialEq, Eq, Serialize, Deserialize)]
pub enum StapleType {
    GrainBread,     // 小麦→面包
    GrainRice,      // 水稻→米饭
    GrainPorridge,  // 谷物→粥/糊
    Tuber,          // 根茎类→薯/芋
```

```rust
fn derive_meat_role(core: &CultureCoreParams, biome: &BiomeParams) -> MeatRole {
    // 首要决定因素：环境——草原/畜牧区→肉为核心
    if biome.is_grassland || biome.is_pastoral {
        return MeatRole::Central;
    }
    // 高宗教性→可能走向素食（宗教禁肉戒律——具体禁忌由信仰系统定义）
    // 文化系统只决定"肉的常规地位"
    if core.religiosity > 0.7 {
        return MeatRole::Occasional;
    }
    // 高放纵度→肉为常规享受
    if core.indulgence > 0.6 {
        return MeatRole::Regular;
    }
    MeatRole::Regular
}

fn derive_alcohol_stance(core: &CultureCoreParams) -> AlcoholStance {
    // 高宗教性→可能禁止（但——具体由信仰系统决定）
    // 文化提供倾向，信仰提供具体约束
    if core.religiosity > 0.7 && core.indulgence < 0.3 {
        return AlcoholStance::Discouraged;
    }
    if core.indulgence > 0.6 {
        return AlcoholStance::Accepted; // 享受生活包含饮酒
    }
    if core.long_term_orientation > 0.7 {
        return AlcoholStance::Ritualistic; // 限于仪式使用——"喝酒误事"
    }
    AlcoholStance::Accepted
}

fn derive_dining_style(core: &CultureCoreParams) -> DiningStyle {
    if (1.0 - core.individualism) > 0.6 {
        DiningStyle::CommunalShared // 集体→共食——共享大盘菜
    } else if core.individualism > 0.6 {
        DiningStyle::IndividualPlated // 个人→分食
    } else {
        DiningStyle::CommunalFromCommon
    }
}

fn derive_spice_preference(core: &CultureCoreParams, biome: &BiomeParams) -> f32 {
    let base = if biome.is_tropical { 0.7 } else { 0.3 }; // 热带=香料产地
    (base + core.indulgence * 0.3 + core.openness_to_outsiders * 0.1 + core.artistry * 0.1)
        .clamp(0.0, 1.0)
}
```

## 四、FertilityNorms —— 生育文化规范

### 4.1 数据结构

```rust
/// 文化生育规范 —— 第一层派生参数
/// 
/// 从核心参数派生。输入到生命 012（繁衍系统）的生育行为调制。
/// 
/// 消费者：生命 012（繁衍系统）· 世界生成 P8（人口生成——年生育率+初婚年龄）
#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct FertilityNorms {
    /// 理想家庭规模的子女数
    pub ideal_family_size: (u32, u32),  // (min, max)

    /// 非婚生子女的社会污名程度
    /// 0.0 = 无污名
    /// 1.0 = 严重社会排斥
    pub illegitimacy_stigma: f32,

    /// 性别偏好——对子女性别的偏好
    /// -1.0 = 强烈偏好女孩
    ///  0.0 = 无偏好
    ///  1.0 = 强烈偏好男孩
    pub sex_preference: f32,

    /// 婚姻压力强度——结婚行为受到的社会期待
    /// 0.0 = 无婚姻压力（长期单身完全正常）
    /// 1.0 = 极强压力（适龄不婚=家庭耻辱）
    pub marriage_pressure: f32,
}

impl FertilityNorms {
    pub fn derive_from(core: &CultureCoreParams) -> Self {
        Self {
            ideal_family_size: derive_ideal_family_size(core),
            illegitimacy_stigma: derive_illegitimacy_stigma(core),
            sex_preference: derive_sex_preference(core),
            marriage_pressure: derive_marriage_pressure(core),
        }
    }

    /// 年生育率 —— 从理想家庭规模反推
    ///
    /// 假设生殖窗口 = menopause_age - marriage_age_typical。
    /// 消费方：世界生成 P8 Phase B Pass 4（已故子女生成）
    pub fn annual_fertility_rate(&self) -> f32 {
        let avg_children = (self.ideal_family_size.0 + self.ideal_family_size.1) as f32 / 2.0;
        let reproductive_years: f32 = 25.0; // ~40岁末孕 - ~15岁初孕 ≈ 25年窗口
        (avg_children / reproductive_years).clamp(0.02, 0.5)
    }

    /// 典型初婚年龄 —— 从 CoreParams 派生
    ///
    /// 返回 (女性典型初婚年龄, 男性典型初婚年龄)。
    /// long_term_orientation 高 → 晚婚（延迟满足）。
    /// power_distance 高 → 略早婚（包办婚姻趋向）。
    /// 消费方：世界生成 P8 Phase B Pass 1（祖先创建时的父母出生年份估算）
    pub fn marriage_age_typical(core: &CultureCoreParams) -> (u8, u8) {
        let base_female: f32 = 18.0;
        let base_male: f32 = 20.0;
        let delay = core.long_term_orientation * 6.0;     // 最多 +6 岁
        let early = core.power_distance * 3.0;             // 高权力距离 → 早婚，最多 -3 岁
        let female = (base_female + delay - early).clamp(14.0, 30.0) as u8;
        let male = (base_male + delay - early).clamp(16.0, 35.0) as u8;
        (female, male)
    }
}

/// 理想子女数
/// 高集体主义→多子女（"家族需要延续"）
/// 高长期取向→少子女（"质量比数量重要"）
/// 农业社会→多子女（劳动力需求）——通过群系间接体现
fn derive_ideal_family_size(core: &CultureCoreParams) -> (u32, u32) {
    let base = if (1.0 - core.individualism) > 0.6 { 4.0 } else { 2.0 };
    let adjusted = base * (1.0 - core.long_term_orientation * 0.5) // 长线→少生精养
                        * (1.0 + core.militarism * 0.5); // 军事→需要更多人口
    let mid = adjusted.clamp(1.0, 8.0) as u32;
    (mid.max(1) - 1, mid + 1)
}
```

**注意**：`marriage_pressure` 使用 `NormId::MarriagePressure` 对应的规范（在 NPC 模块中已有定义）。本文档在此定义 culture-level 的 baseline 值——个体 NPC 的 internalization 在此基础上 ±0.15 偏移。

### 派生公式

```rust
impl RelationshipNorms {
    pub fn derive_from(core: &CultureCoreParams) -> Self {
        // —— 数量结构 ——
        // 高 power_distance + 高 religiosity → 一夫多妻（传统等级社会）
        // 高 individualism → 一夫一妻连续（个人选择）
        // 低 individualism + 低 power_distance → 群婚可能（平等集体社群）

        let hierarchy = core.power_distance * core.religiosity;
        let monogamous_default = core.individualism > 0.5 || hierarchy < 0.3;

        let polygyny_acceptance = (hierarchy * 0.8 + core.militarism * 0.2  // 军事→男性稀缺→多妻
            - core.individualism * 0.4).clamp(0.0, 1.0);

        let polyandry_acceptance = ((1.0 - core.power_distance) * 0.3  // 平等社会中可能出现
            + core.long_term_orientation * 0.1  // 资源稀缺→兄弟共妻（经济理性）
            - core.religiosity * 0.3).clamp(0.0, 0.3);  // 上限 0.3——全球罕见

        let polyamory_acceptance = (core.individualism * 0.5  // 个人自由
            + core.indulgence * 0.3
            - core.religiosity * 0.6
            - core.uncertainty_avoidance * 0.2).clamp(0.0, 0.4);

        let group_marriage_acceptance = ((1.0 - core.individualism) * 0.3  // 集体社会
            + (1.0 - core.power_distance) * 0.2  // 平等
            - core.religiosity * 0.5).clamp(0.0, 0.2);

        let max_spouses = if polygyny_acceptance > 0.5 || polyandry_acceptance > 0.3 {
            Some(2 + (core.power_distance * 6.0) as u8)  // 2-8 人
        } else {
            None  // 一夫一妻，无多配偶
        };

        // —— 形成方式 ——
        let arranged_marriage_tendency = (core.power_distance * 0.7
            + (1.0 - core.individualism) * 0.3
            + core.uncertainty_avoidance * 0.2).clamp(0.0, 1.0);

        let contractual_marriage_acceptance = (core.individualism * 0.4
            + core.long_term_orientation * 0.3  // 长期取向→短期契约
            - core.religiosity * 0.5).clamp(0.0, 0.6);

        // —— 性别与取向 ——
        let homosexuality_acceptance = (core.individualism * 0.6  // 个人自由
            + core.openness_to_outsiders * 0.2  // 对差异的包容
            - core.religiosity * 0.5
            - core.uncertainty_avoidance * 0.3  // 害怕差异
            - core.power_distance * 0.3  // 等级社会→性别角色严格
            ).clamp(0.0, 1.0);

        let same_sex_marriage_legal = if homosexuality_acceptance > 0.7 {
            homosexuality_acceptance  // 几乎等同于异性婚姻
        } else if homosexuality_acceptance > 0.4 {
            homosexuality_acceptance * 0.5  // 民事结合但不叫婚姻
        } else {
            0.0  // 不承认
        };

        // —— 生物与社会 ——
        let conjugal_birth_norm = (core.religiosity * 0.6
            + core.power_distance * 0.3  // 家族控制
            + core.uncertainty_avoidance * 0.2  // 社会秩序
            - core.individualism * 0.3).clamp(0.0, 1.0);

        // —— 离婚 ——
        let divorce_acceptance = (core.individualism * 0.5
            + core.indulgence * 0.2
            - core.religiosity * 0.6
            - core.uncertainty_avoidance * 0.3
            - core.long_term_orientation * 0.2  // 长期承诺→不离婚
            ).clamp(0.0, 1.0);

        // —— 居住模式 ——
        let residence_pattern = derive_residence_pattern(core);

        Self {
            monogamous_default,
            polygyny_acceptance, polyandry_acceptance,
            polyamory_acceptance, group_marriage_acceptance,
            max_spouses,
            arranged_marriage_tendency, contractual_marriage_acceptance,
            homosexuality_acceptance, same_sex_marriage_legal,
            conjugal_birth_norm,
            divorce_acceptance,
            residence_pattern,
        }
    }
}

fn derive_residence_pattern(core: &CultureCoreParams) -> ResidencePattern {
    // v1.1 扩展为 4 项公式——所有权重归一化到 1.0，增加维度覆盖
    let patrilocal_score = core.power_distance * 0.50
        + core.uncertainty_avoidance * 0.25
        + (1.0 - core.individualism) * 0.15
        + core.militarism * 0.10;
    let matrilocal_score = (1.0 - core.power_distance) * 0.35
        + (1.0 - core.militarism) * 0.30
        + core.individualism * 0.20
        + core.openness_to_outsiders * 0.15;
    let neolocal_score = core.individualism * 0.50
        + core.openness_to_outsiders * 0.20
        + (1.0 - core.uncertainty_avoidance) * 0.20
        + core.indulgence * 0.10;
    let ambilocal_score = core.openness_to_outsiders * 0.50
        + (1.0 - core.uncertainty_avoidance) * 0.30
        + core.indulgence * 0.20;

    // 加权随机（由种子 + 核心参数确定性推导，非运行时随机）
    // 实现时使用 hash(culture_id, "residence") 作为确定性选择
    let max_score = patrilocal_score.max(matrilocal_score)
        .max(neolocal_score).max(ambilocal_score);

    if max_score == patrilocal_score { ResidencePattern::Patrilocal }
    else if max_score == matrilocal_score { ResidencePattern::Matrilocal }
    else if max_score == neolocal_score { ResidencePattern::Neolocal }
    else { ResidencePattern::Ambilocal }
}
```

**注意**：所有系数标记 `[TUNING]`。RelationshipNorms 使用扁平字段而非引用 NPC 模块的 `RelationshipStructure` 枚举——保持文化系统对 NPC 模块的零依赖。NPC 模块消费时自行映射到枚举。

## 六、荣耀权重（honor_weight）

`honor_weight` 是第一层派生参数——从 7 个核心参数的组合中涌现，不新增核心参数：

```rust
/// 荣誉权重 —— 该文化中"荣誉/面子"在社交地位中的重要性
/// 
/// 从多个核心参数的特定组合中涌现——不是独立参数。
/// 消费者的统一查询接口（而非让每个模块独立计算自己的"荣誉版本"）。
pub fn honor_weight(core: &CultureCoreParams) -> f32 {
    let raw = (1.0 - core.individualism) * 0.25   // 集体主义 → 集体荣誉
            + core.power_distance * 0.20           // 等级分明 → 地位荣誉
            + core.uncertainty_avoidance * 0.20    // 传统主义 → 祖辈荣誉
            + core.competition_orientation * 0.15   // 竞争 → 名誉竞争
            + core.militarism * 0.10               // 军事 → 武力捍卫荣誉
            + (1.0 - core.openness_to_outsiders) * 0.10; // 排外 → 外人不可信
    raw.clamp(0.0, 1.0)
}
```

**消费者**：历史系统（立碑传统权重 ×2.0 条件）、NPC（`grudge_duration` 修正）、权力系统（征服后的 legitimacy 恢复速度）、战斗系统（投降是否视作耻辱）。
