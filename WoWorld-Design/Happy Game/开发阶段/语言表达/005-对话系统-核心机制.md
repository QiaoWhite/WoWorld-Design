## 一、Conversation——多参与者对话容器

### 1.1 核心定义

```rust
/// 一场对话——不限于两个人
/// 一对一只是 participants.len() == 2 的特例
pub struct Conversation {

    pub participants: Vec<Participant>,

    /// 发言模式——决定"谁下一个说话"

    pub history: Vec<TurnRecord>,

    pub social_field: SocialField,

    /// 对话的物理上下文
    pub setting: ConversationSetting,

    /// 对话开始时间
    pub started_at: GameTimestamp,
}

pub struct ConversationSetting {
    /// 对话地点
    pub location: LocationId,
    /// 物理布局——围坐在桌边/站在广场上/面对面
    pub layout: ConversationLayout,
    /// 是否有旁观者
    pub has_bystanders: bool,
}

pub enum ConversationLayout {
    FaceToFace,        // 面对面——2-3 人
    AroundTable,       // 围坐——4-8 人
    OpenAudience,      // 讲台/舞台——演讲者对听众
    StandingCircle,    // 站圈——非正式聚集
    RitualPositions,   // 仪式站位——婚礼/法庭
```

### 1.2 Participant——参与者

```rust
pub struct Participant {
    pub entity_id: EntityId,  // NpcId 或 PlayerId

    /// 角色——决定发言权权限
    pub role: ParticipantRole,

    /// 影响发言影响力（高地位者发言→SocialField 变化更大）

    /// 主讲人——持有长时间发言权

}

pub struct AttentionFocus {

    /// 正在旁听的对话列表（可以听到但没参与）
    pub eavesdropping: Vec<ConversationId>,
    /// 旁听信息的清晰度——距离越远，听到的越少
```

## 二、TurnMode——发言权分配

### 2.1 四种模式

```rust
pub enum TurnMode {
    /// 自由交谈——谁想说话就说话
    /// 适合: 3-8 人闲聊、酒馆桌、冒险小队
    FreeForm {
        /// 抢话阈值——当前说话者说完后，谁先开口
        /// 由 personality.extraversion × willingness × authority 驱动
        interrupt_threshold: f32,
    },

    /// 轮流发言——主持人分配发言权
    /// 适合: 议会、法庭、正式会议
    Moderated {
        moderator: EntityId,
        /// 排队等待发言的人
        queue: VecDeque<EntityId>,
    },

    /// 演讲——一个人对多人说话
    /// 适合: 公共演讲、布道、战前动员、国王诏令
    Speech {
        speaker: EntityId,
        /// 听众是否可以打断（提问/嘘声/欢呼）
        audience_can_interrupt: bool,
        /// 打断的类型限制
        allowed_interruptions: Vec<InterruptionType>,
    },

    /// 仪式对话——严格按脚本走
    /// 适合: 婚礼誓言、法庭宣誓、结拜、加冕
    Ritual {
        steps: Vec<RitualStep>,
        current_step: usize,
    },
}

pub enum InterruptionType {
    Question,      // 提问
    Cheer,         // 欢呼
    Boo,           // 嘘声
    WalkOut,       // 离场
}

pub struct RitualStep {
    pub speaker: EntityId,
    pub text_template_id: u32,
    pub params: HashMap<String, String>,
    pub duration_seconds: f32,
}
```

### 2.2 TurnAllocator——发言权分配器

```rust
pub struct TurnAllocator;

impl TurnAllocator {
    /// 自由交谈模式——评估每个参与者的发言驱动力
    pub fn allocate_freeform(
        participants: &[Participant],
        social_field: &SocialField,
        current_speaker: Option<EntityId>,
    ) -> Option<EntityId> {
        let candidates: Vec<TurnCandidate> = participants.iter()
            .filter(|p| p.role == ParticipantRole::Active)
            .filter(|p| Some(p.entity_id) != current_speaker)  // 不和自己抢
            .map(|p| TurnCandidate {
                participant_id: p.entity_id,
                speak_drive: Self::speak_drive(p, social_field),
            })
            .filter(|c| c.speak_drive > 0.15)  // 阈值——不想说话的人沉默
            .collect();

        if candidates.is_empty() { return None; }

        // 最高驱动力者获得发言权
        candidates.into_iter()
            .max_by(|a, b| a.speak_drive.partial_cmp(&b.speak_drive).unwrap())
            .map(|c| c.participant_id)
    }

    /// 个体的发言驱动力
    fn speak_drive(participant: &Participant, social_field: &SocialField) -> f32 {
        participant.willingness_to_speak * 0.35
            + participant.authority_on_topic * 0.30
            + personality_drive(participant.extraversion) * 0.20
            + emotional_urgency(participant.emotion) * 0.10
            + topic_novelty_bonus(social_field) * 0.05
    }
}

struct TurnCandidate {
    participant_id: EntityId,
    speak_drive: f32,
}
```

## 三、DialogueIntent——NPC 主动对话驱动

### 3.1 五种驱动类型

```rust
/// NPC 在 GOAP/行为树战略层每 5-30 秒评估——是否需要主动和人说话
pub struct DialogueIntent {
    pub intent_type: DialogueIntentType,
    /// 优先级——多个意图竞争，行为树选择最高优先级
    pub priority: f32,
    /// 目标说话对象（None = 最近的合适对象）
    pub target: Option<EntityId>,
    /// 意图携带的参数
    pub topic_params: HashMap<String, String>,
    /// 过期时间——超过此时间意图自动取消
    pub expires_at: GameTimestamp,
}

pub enum DialogueIntentType {
    /// GOAP 驱动——NPC 有个目标需要口头沟通才能推进
    /// "铁匠想卖剑 → 找最近的未武装 NPC → 发起推销对话"
    GoalDriven {
        goal_id: GoalId,
        /// "persuade_purchase" / "request_help" / "warn_danger" / "report_crime"
        action: String,
    },

    /// 情感驱动——高唤醒情绪需要社会出口
    /// "极其愤怒 → 找朋友倾诉" / "欣喜若狂 → 找人分享"
    Emotional {
        emotion: Emotion,
        intensity: f32,     // 0-1，来自情绪引擎的 arousal
        /// 导致该情绪的记忆 ID（如果有）
        cause_memory_id: Option<MemoryId>,
    },

    /// 关系维护——太久没和某个重要的人互动
    SocialMaintenance {
        target: EntityId,
        days_since_last_interaction: u32,
    },

    /// 信息传播——刚得知值得传播的记忆
    SpreadInformation {
        memory_id: MemoryId,
        /// 信息紧迫度——"蛮族入侵了！" > "村口老王家的鸡又跑了"
        urgency: f32,
    },

    /// 对玩家反应——观察到玩家做了值得评论的事
    ReactToPlayer {
        /// "killed_dragon" / "stole_item" / "helped_villager" / "won_duel"
        observed_action: String,
        /// -1.0(极其负面) → +1.0(极其正面)
        valence: f32,
    },
}
```

### 3.2 意图优先级公式

```rust
impl DialogueIntent {
    /// 计算意图优先级——用于行为树选择"
    pub fn compute_priority(&self, npc: &NpcData) -> f32 {
        let base = match &self.intent_type {
            DialogueIntentType::GoalDriven { .. } => 0.6,
            DialogueIntentType::Emotional { intensity, .. } => *intensity * 0.8,
            DialogueIntentType::SocialMaintenance { days_since_last_interaction, .. } => {
                (days_since_last_interaction.min(&14) / 14.0)  // 越久权重越高
            }
            DialogueIntentType::SpreadInformation { urgency, .. } => *urgency,
            DialogueIntentType::ReactToPlayer { valence, .. } => valence.abs() * 0.9,
        };

        // 目标近距加成——说话对象在身边时优先级更高
        let proximity_bonus = self.target_proximity_bonus(npc);

        base * proximity_bonus
    }
}
```

### 4.2 TopicSelector——话题选择

```rust
/// 话题调度器——决定"当前对话中可以聊什么"
pub struct TopicSelector {
    /// 话题类别的基础权重配置
    category_weights: HashMap<TopicCategory, f32>,
}

impl TopicSelector {
    /// 从 NPC 状态和对话上下文生成预设话题列表
    pub fn select_topics(
        &self,
        npc: &NpcSnapshot,
        ctx: &DialogueContext,
        count: usize,
    ) -> Vec<ConversationTopic> {
        let mut candidates: Vec<(ConversationTopic, f32)> = Vec::new();

        // 从 NPC 记忆生成话题
        for memory in &ctx.recent_memories {
            let weight = self.compute_topic_weight(memory, npc, ctx);
            if weight > 0.1 {
                candidates.push((ConversationTopic::Preset {
                    topic_id: TopicId::from_memory(memory.id),
                    label: self.generate_label(memory),
                    category: self.classify(memory),
                    relevance: weight,
                    source_memory: Some(memory.id),
                }, weight));
            }
        }

        // 从天气/环境生成话题
        if let Some(weather_topic) = self.weather_topic(ctx) {
            candidates.push((weather_topic, 0.3));  // 天气 = 低权重但总是可用
        }

        // 从 NPC 职业/技能生成话题
        let prof_topics = self.profession_topics(npc);
        for (topic, weight) in prof_topics {
            candidates.push((topic, weight));
        }

        // Alias Method 加权采样——O(n) 建表，O(1) 采样
        self.weighted_sample(&candidates, count)
    }

    /// 单个话题的权重计算
    fn compute_topic_weight(
        &self,
        memory: &EventMemory,
        npc: &NpcSnapshot,
        ctx: &DialogueContext,
    ) -> f32 {
        let category_base = self.category_weights
            .get(&self.classify(memory))
            .copied()
            .unwrap_or(0.3);

        // 新鲜度——越近越高
        let recency = (1.0 - (memory.age_days() as f32 * 0.05)).max(0.1);

        // 记忆的影响力
        let impact = memory.impact_score;

        // 个性修正
        let personality_mod = self.personality_weight_modifier(npc, memory);

        // 关系修正——和当前对话对象的关系影响话题选择
        let relationship_mod = self.relationship_weight_modifier(ctx, memory);

        category_base * recency * impact * personality_mod * relationship_mod
    }

    /// 个性对话题权重的修正
    fn personality_weight_modifier(&self, npc: &NpcSnapshot, memory: &EventMemory) -> f32 {
        let mut modifier = 1.0;
        // 高开放性 → 知识/学术类话题 +30%
        // 高外向性 → 闲聊/八卦类话题 +30%
        // 高宜人性 → 情感/关系类话题 +20%
        // 低开放性 → 知识/学术类话题 -20%
        modifier
    }

    /// 从大日志条目生成对话话题
    pub fn from_journal(
        &self,
        entry: &JournalEntry,
        npc: &NpcData,
    ) -> ConversationTopic {
        ConversationTopic::JournalReference {
            entry_id: entry.id,
            summary: entry.summary.clone(),
            tags: entry.tags.clone(),
            source: entry.acquisition_method.into(),
        }
    }
}
```

### 4.3 TopicResolver——话题解析与回应生成

```rust
/// 话题解析器——给定话题和 NPC，产出回应文本 + 新选项
pub struct TopicResolver;

impl TopicResolver {
    pub fn resolve(
        topic: &ConversationTopic,
        speaker: &NpcSnapshot,
        ctx: &DialogueContext,
        generator: &TextGenerator,
    ) -> NpcResponse {
        match topic {
            ConversationTopic::Preset { source_memory, .. } => {
                // NPC 从自己的记忆中提取相关回忆 → 生成回应
                self.resolve_preset_topic(topic, speaker, ctx, generator)
            }
            ConversationTopic::JournalReference { entry_id, tags, source, .. } => {
                // 玩家提了一个 NPC 可能不知道的话题
                // → 1. 检查 NPC 自己的记忆是否匹配
                // → 2. 检查 NPC 的技能/知识是否相关
                // → 3. 完全不匹配 → 好奇/怀疑/不知的回应
                self.resolve_journal_topic(topic, speaker, ctx, generator)
            }
            ConversationTopic::FreeTextTopic { text, extracted_entities, confidence } => {
                // 自由文本 → 提取的实体 → 尽量匹配到预设话题
                // 匹配成功 → 走预设话题管道
                // 匹配失败(低置信度) → NPC "不太明白你的意思" + 提供备选选项
                self.resolve_freeform_topic(topic, speaker, ctx, generator)
            }
        }
    }

    /// NPC 不知道话题 —— 不失败，反而是"活人感"的来源
    fn resolve_journal_topic(
        &self,
        topic: &ConversationTopic,
        speaker: &NpcSnapshot,
        ctx: &DialogueContext,
        generator: &TextGenerator,
    ) -> NpcResponse {
        if let ConversationTopic::JournalReference { tags, source, .. } = topic {
            // ① 检查 NPC 是否就是信息来源
            if let TopicSource::DirectConversation = source {
                if self.speaker_is_original_source(speaker, topic) {
                    // "哦，我跟你说过这个？对——我爷爷那辈传下来的。你还想听更多？"
                    return self.respond_as_source(topic, speaker, ctx, generator);
                }
            }

            // ② 检查 NPC 自己的记忆——有没有匹配标签的事件？
            let npc_memories = self.find_matching_memories(speaker, tags);
            if !npc_memories.is_empty() {
                // "火龙谷？我当然知道。四十年前我还亲眼见过一次。"
                return self.respond_with_knowledge(topic, speaker, &npc_memories, ctx, generator);
            }

            // ③ 检查 NPC 的技能——有没有相关知识？
            let npc_knowledge = self.find_related_knowledge(speaker, tags);
            if !npc_knowledge.is_empty() {
                // 学者 NPC: "火龙谷的传说有好几个版本。最早见于《苍山异兽志》……"
                return self.respond_with_expertise(topic, speaker, &npc_knowledge, ctx, generator);
            }

            // ④ 完全不知道——好奇/怀疑/诚实的回应
            // "火龙？真的假的？我这辈子没见过——你亲眼看见的？"
            return self.respond_with_curiosity(topic, speaker, ctx, generator);
        }
        // fallback
        self.default_response(speaker, ctx, generator)
    }
}

/// NPC 回应结构

    pub player_options: Vec<PlayerOption>,
}

pub struct PlayerOption {

    pub tone: Option<Tone>,
}

pub enum Tone { Positive, Neutral, Negative, Aggressive, Playful }
```

## 七、完整对话流程示例

### 7.1 NPC 主动对话

```
① NPC 行为树每 10 秒评估 DialogueIntent 生成条件
   → 铁匠老王: 刚得知玩家屠龙 → 生成 ReactToPlayer intent (priority 1.01)

② 玩家进入铁匠铺 (L1 激活)
   → 老王的 DialogueIntent 排在行为队列首位
   → 老王走近玩家（社交距离 1.5m）

③ Godot: 弹出提示 "[铁匠老王] 想和你说句话"

④ 玩家按交互键 → Conversation::start()

⑤ 调用者在外部收集 DialogueContext:
   → npc_snapshot (老王: agreeableness=0.4, extraversion=0.5)
   → relationships (老王→玩家: trust=0.3)
   → recent_memories (包含屠龙事件的二阶传播记忆)
   → known_languages (中原通用语)
   → player_journal_entries (玩家的全部大日志)
   → item_names (玩家背包物品名)
   → location_name ("苍山城铁匠铺")

⑥ TopicSelector::from_intent(intent, npc) → 候选话题
   TopicSelector::select_topics(npc, ctx, 4) → 4 个预设话题

⑦ 开场——PhaticLayer::Opening → "下雨天还出来？"

⑧ 展示选项:
   [A] "顺便看看。"                  (Phatic)
   [B] "听说城门那边有条龙……"       (Preset: 屠龙事件 ← 权重最高)
   [C] "上次那把铁剑……"            (Preset: 交易记忆)
   [D] 🔍 [搜索大日志……]           (搜索框入口)

⑨ 玩家选择 → TopicResolver::resolve() → NpcResponse

⑩ 对话持续 3-5 轮 → 话题枯竭或玩家告辞 → PhaticLayer::Closing → 对话结束
```
