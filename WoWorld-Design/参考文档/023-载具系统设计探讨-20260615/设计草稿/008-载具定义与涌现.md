## 一、三层来源模型

```
VehicleArchetype (TOML, 人手写, ~20-30个)
  │  定义"角色"——这个载具是用来干什么的
  │  不绑定任何文化/群系/技术的具体参数
  │
  ├─ × CultureCoreParams → CulturalShipbuildingTradition
  │     文化核心参数 → 造船传统的连续参数
  │     (长度比/吃水深/骨架风格/帆装/船首形状...)
  │
  ├─ × TechnologyProfile → 技术上限约束
  │     技术等级 → 最大船长/水密隔舱/龙骨技术
  │
  ├─ × BiomeParams → 材料可用性
  │     群系 → 可用木材/金属/纤维
  │
  └─ × seed → 微小随机变体
        └─ → VehicleDef (涌现, ~200-500个, 一次性确定性计算)
               └─ → VehicleState (运行时, ~3K-10K个)
```

## 二、VehicleArchetype —— TOML 原型

```toml
# archetypes/water/small_fishing_boat.toml

[archetype]
id = "core.water.fishing_boat_small"
name_key = "vehicle.archetype.fishing_boat_small"
domain = "Water"

[size_class]
base = "Tiny"
# 文化可能将相同原型变成 Small 或保持 Tiny

[capacity]
cargo_kg_range = [50, 500]
cargo_liters_range = [200, 1500]
passengers_range = [1, 5]

[propulsion]
# 允许的动力类型及其文化适配
allowed_kinds = ["MusclePowered", "WindPowered"]
# 基准速度范围——文化和技术在其中变形
base_speed_range_ms = [1.5, 5.0]

[crew_roles]
[[crew_roles.helm]]
label_key = "crew.helm"
skill = "Navigation"
min_count = 1
critical = true

[[crew_roles.fisher]]
label_key = "crew.fisher"
skill = "Fishing"
min_count = 1
critical = false
# count 可变——文化可能加一个"学徒"位

[construction]
base_days = [1, 7]
site = "Waterfront"
min_water_depth_m_range = [0.3, 2.0]

[damage]
storm_resistance_range = [0.05, 0.3]
ice_resistance_range = [0.0, 0.15]

[param_bounds]
length_m = [2.0, 10.0]
beam_m = [0.8, 3.5]
draft_m = [0.2, 1.5]
```

## 三、CulturalShipbuildingTradition —— 文化对载具的塑造力

```rust
/// 文化造船传统——从 CultureCoreParams + TechnologyProfile + BiomeParams 派生
/// 所有参数都是连续值 0-1，代表文化偏好
/// 不是"北欧文化=长船"的硬编码——参数组合涌现出船型特征
#[derive(Debug, Clone)]
pub struct CulturalShipbuildingTradition {
    // ── 结构偏好 ──
    /// 长宽比倾向: 0=宽胖(稳), 1=细长(快)
    pub length_to_beam_ratio: f32,
    /// 吃水深比: 0=极浅吃水(浅水文化), 1=深吃水(远洋文化)
    pub hull_depth_ratio: f32,
    /// 骨架风格: 0=轻骨架(搭接式·北欧), 0.5=混合, 1=重骨架(中式·骨架先建)
    pub frame_material_tradition: f32,
    /// 船板拼接: 0=搭接(clinker), 0.5=平接(carvel), 1=榫接
    pub planking_style: f32,

    // ── 动力偏好 ──
    /// 帆数倾向: 0=单帆, 1=多帆
    pub sail_count_tendency: f32,
    /// 桨依赖: 0=纯帆, 1=重桨
    pub oar_reliance: f32,

    // ── 审美偏好 ──
    pub bow_shape: f32,             // 0=垂直艏, 0.5=倾斜艏, 1=勺形艏
    pub stern_shape: f32,           // 0=双头(北欧长船可进退), 0.5=方艉, 1=椭圆艉
    pub decoration_density: f32,    // 0=朴素, 1=繁复
    pub figurehead_tradition: f32,  // 0=无, 0.5=船首像, 1=复杂艏饰

    // ── 技术约束 ──
    pub max_vessel_length_m: f32,
    pub has_watertight_compartment: bool,
    pub keel_technology: KeelTech,

    // ── 材料偏好 ──
    pub preferred_hull_material: ItemDefId,   // 橡木/松木/竹/...
    pub preferred_sail_material: Option<ItemDefId>,
}

#[derive(Debug, Clone, Copy, PartialEq, Eq)]
pub enum KeelTech {
    None,       // 无龙骨——平底船
    Shallow,    // 浅龙骨
    Deep,       // 深龙骨——远洋船
}

impl CulturalShipbuildingTradition {
    /// 从文化核心参数 + 技术 + 群系 派生
    pub fn derive(
        culture: &CultureCoreParams,
        tech: &TechnologyProfile,
        biome: &BiomeParams,
    ) -> Self {
        CulturalShipbuildingTradition {
            // 长宽比: 个人主义文化→更长更快, 集体主义→更宽更稳
            length_to_beam_ratio: lerp(
                0.3, 0.8,
                culture.individualism,
            ),
            // 吃水深: 不确定性规避高→近海浅水安全, 低→远洋深吃水
            hull_depth_ratio: lerp(
                0.2, 0.9,
                1.0 - culture.uncertainty_avoidance,
            ),
            // 骨架: 竞争导向高→轻骨架(需要速度), 低→重骨架(耐久)
            frame_material_tradition: lerp(
                0.7, 0.2,
                culture.competition_orientation,
            ),
            // 帆数: 开放性高→试新事物→多帆
            sail_count_tendency: lerp(
                0.1, 0.9,
                culture.openness_to_outsiders,
            ),
            // 桨依赖: militarism 高→需要桨(无风时保持机动)
            oar_reliance: lerp(
                0.0, 0.8,
                culture.militarism,
            ),
            // 审美: artistry 高→装饰繁复
            decoration_density: lerp(
                0.05, 0.95,
                culture.artistry,
            ),
            // 技术约束
            max_vessel_length_m: tech.metallurgy * 80.0 + tech.construction * 40.0,
            has_watertight_compartment: tech.construction > 0.4,
            keel_technology: match tech.construction {
                t if t < 0.2 => KeelTech::None,
                t if t < 0.5 => KeelTech::Shallow,
                _ => KeelTech::Deep,
            },
            // 材料: 群系决定可用木材
            preferred_hull_material: select_wood(biome, culture),
            preferred_sail_material: select_fiber(biome, culture),
            // ...其余字段类似地从 CultureCoreParams 派生
            ..Default::default()
        }
    }
}
```

## 四、VehicleDef 的确定性计算

```rust
/// 从原型 + 文化 + 种子 → 载具类型定义
/// 纯函数——同输入永远同输出
fn concretize_vehicle_def(
    archetype: &VehicleArchetype,
    tradition: &CulturalShipbuildingTradition,
    tech: &TechnologyProfile,
    seed: u64,
) -> VehicleDef {
    let rng = seeded_rng(seed, archetype.id);

    // 1. 几何——从文化传统 + 原型边界插值
    let length = lerp_in_range(
        archetype.param_bounds.length_m,
        tradition.max_vessel_length_m.min(40.0),
        tradition.length_to_beam_ratio,
    );
    let beam = length / lerp(3.0, 8.0, tradition.length_to_beam_ratio);
    let draft = length * lerp(0.05, 0.25, tradition.hull_depth_ratio);

    // 2. 航速——物理公式，非查表
    let base_speed = estimate_hull_speed_ms(length, beam, &archetype.propulsion);

    // 3. 容量——从几何派生
    let (cargo_kg, cargo_liters) = estimate_cargo_capacity(
        length, beam, draft, archetype.size_class
    );

    // 4. 船员岗位——原型定义角色，文化本地化名称
    let crew_slots = archetype.crew_roles.iter().map(|role| {
        CrewSlotDef {
            label_key: role.label_key.clone(),  // 本地化由语言表达系统处理
            required_skill: role.skill,
            count: role.min_count + rng.gen_range(0..=1),
            critical: role.critical,
            // ...
        }
    }).collect();

    // 5. 损伤抗性——从结构派生
    let storm_resistance = derive_storm_resistance(length, tradition.keel_technology);
    let ice_resistance = derive_ice_resistance(tradition.planking_style);

    // 6. 外观——纯文化派生
    let aesthetics = VehicleAesthetics {
        bow_shape: tradition.bow_shape + rng.noise_01() * 0.1,
        stern_shape: tradition.stern_shape,
        decoration_level: tradition.decoration_density * culture.artistry,
        preferred_hull_material: tradition.preferred_hull_material,
        // ...
    };

    VehicleDef {
        def_id: VehicleDefId::from_parts(archetype.id, tradition.culture_id, seed),
        archetype: archetype.id,
        culture_id: tradition.culture_id,
        domain: archetype.domain,
        geometry: VehicleGeometry { length_m: length, beam_m: beam, draft_m: draft, /*...*/ },
        capacity: VehicleCapacity { cargo_kg, cargo_liters, passengers },
        propulsion: concretize_propulsion(&archetype.propulsion, tradition, tech),
        crew_slots,
        damage_profile: VehicleDamageProfile { storm_resistance, ice_resistance, /*...*/ },
        physical: derive_physical_profile(length, cargo_kg, &propulsion),
        aesthetics,
        // ...
    }
}
```
