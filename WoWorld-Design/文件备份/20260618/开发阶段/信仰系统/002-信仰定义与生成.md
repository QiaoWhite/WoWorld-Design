## 一、FaithDefinition——完整信仰定义

### 1.1 顶层结构

```rust
/// FaithDefinition = 一个具体的宗教实例的完整定义。
/// 世界生成 P2.5 从种子生成——纯数据，不需要 NPC 或聚落数据。
/// 种子生成数量：通常 20-200 个（每个文化按 religiosity 生成 0-3 个）。
/// 存储：信仰系统 SoA 列式存储，~500 bytes/faith × ~200 = ~100KB 总量。
#[derive(Clone, Serialize, Deserialize)]
pub struct FaithDefinition {
    /// 全局唯一标识符——扁平 u32
    pub id: FaithId,

    /// 信仰名称——内名(音系规则生成) + 外名(5源加权随机)
    pub name: FaithName,

    /// 神学参数——10 连续值，所有特征从此派生
    pub theology: FaithTheology,

    /// 神祇/崇拜对象列表
    /// 单神信仰：1 个条目（importance=1.0）
    /// 多神信仰：多个条目，各有 domains 和 importance
    /// 非神论传统：空（如某些佛教传统），或包含"觉悟者"/"圣贤"（importance 较低）
    pub deities: Vec<DeityDefinition>,

    /// 核心教义——信仰的基本信念主张
    pub doctrines: Vec<Doctrine>,

    /// 禁忌清单——该信仰禁止的行为
    pub taboos: Vec<Taboo>,

    /// 经典文本引用——指向语言表达系统的 ExpressionRef
    pub scriptures: Vec<ScriptureRef>,

    /// 神圣日列表——供 FaithCalendarQuery 返回
    pub holy_days: Vec<HolyDay>,

    /// 斋戒规则——供 FaithCalendarQuery 返回
    pub fasting_rules: Vec<FastingRule>,

    /// 朝圣目标——如果该信仰有朝圣传统
    pub pilgrimage_targets: Vec<PilgrimageTarget>,

    /// 宗教组织结构
    pub institution: InstitutionStructure,

    /// 神职角色定义——该信仰特有的神职类型
    pub clergy_roles: Vec<ClergyRole>,

    /// 对其他信仰的态度
    pub inter_faith_attitude: InterFaithAttitude,

    /// 魔法-宗教关系
    pub magic_relation: MagicReligionRelation,

    /// 繁衍规范——所有权从生命模块 012 迁移至此处
    pub reproduction: ReligiousReproductionNorms,

    /// 神圣建筑参数——供世界生成建筑系统消费
    pub sacred_architecture: SacredArchitectureParams,

    /// 葬仪偏好——供生命系统 DeathCustom 消费
    pub funeral_customs: Option<FuneralCustoms>,

    /// 元数据
    pub origin_culture: CultureId,
    /// 分裂来源——None 表示从种子独立生成
    pub parent_faith: Option<FaithId>,
    /// 创生年份——可为负值（世界历史起点之前）
    pub creation_game_year: i32,
}
```

### 1.3 教义与禁忌

```rust
/// 一条核心教义——信仰的基本信念主张。
pub struct Doctrine {
    /// 简短标题："三位一体"、"轮回"、"原罪"、"八正道"
    pub title: String,
    /// 教义内容——2-3 句的文本描述
    pub description: String,
    /// 教义类别——影响 NPC 行为的具体领域
    pub category: DoctrineCategory,
    /// 对此教义异议的容忍度 (0.0-1.0)
    /// 0.0 = 可讨论/可质疑/可不同解释
    /// 1.0 = 不可质疑的核心信条——违者=异端
    pub disputability: f32,
}

pub enum DoctrineCategory {
    Cosmology,        // 宇宙论——世界如何诞生/终结
    Soteriology,      // 救赎论——如何得救/觉悟/超脱
    Ethics,           // 伦理——何为善、何为恶
    Afterlife,        // 死后——灵魂去向、审判、轮回
    Community,        // 社群——信徒之间的关系、对外态度
    Practice,         // 修行——如何正确进行宗教实践
}

/// 一条禁忌——该信仰禁止的行为。
pub struct Taboo {
    /// 禁忌内容："不食猪肉"、"安息日不可劳作"、"不可拜偶像"
    pub description: String,
    /// 适用范围
    pub scope: TabooScope,
    /// 严重性——违反此禁忌的严重程度 (0.0-1.0)
    /// 0.0 = 轻微不当（"建议不这么做"）
    /// 1.0 = 极重罪（"违者逐出教会/石刑"）
    pub severity: f32,
    /// 禁忌类别
    pub category: TabooCategory,
    /// 违反后的后果类型
    pub consequence: TabooConsequence,
}

pub enum TabooScope {
    AllBelievers,      // 所有信徒
    ClergyOnly,        // 仅神职人员
    DuringHolyDay,     // 仅在神圣日期间
    Initiates,         // 仅入门者（未完成特定仪式的信徒）
    SpecificGender(BiologicalSex),
}

pub enum TabooCategory {
    Dietary,           // 饮食——不吃猪肉/不吃牛肉/素食/斋月
    Dress,             // 着装——头巾/特定颜色/圣徽
    Time,              // 时间——安息日/斋月/每日五次祈祷
    Relationship,      // 关系——同信通婚/禁欲/多妻制
    Ritual,            // 仪式——特定仪式不可省略/不可错误执行
    Speech,            // 言语——不可妄称神名/不可亵渎/不可说谎
}

pub enum TabooConsequence {
    GuiltOnly,              // 仅内心罪咎——无外部惩罚
    SocialDisapproval,      // 社区侧目——影响社交
    RitualImpurity,         // 仪式不洁——需净化仪式
    FormalPenance,          // 正式忏悔——祭司指定的补赎
    Excommunication,        // 逐出教会——切断社群联系
    LegalPenalty,           // 法律惩罚——仅当 religious_law = PolityScope（神权政体）
}
```

### 2.2 从参数派生信仰分类标签

```rust
/// 从 FaithTheology 参数组合派生——纯函数，只用于 UI 分类和文本生成。
/// 不参与任何模拟逻辑。
pub fn derive_faith_label(theology: &FaithTheology) -> FaithLabel {
    if theology.deity_count < 0.05 && theology.ancestor_importance < 0.3 && theology.nature_sacredness < 0.1 {
        return FaithLabel::NonTheistic;  // 无崇拜对象——哲学/伦理传统
    }
    if theology.deity_count > 2.0 && theology.hierarchy_degree > 0.5 {
        return FaithLabel::Polytheism(PolytheismSub::Pantheon);  // 万神殿型多神教
    }
    if theology.deity_count > 2.0 && theology.nature_sacredness > 0.6 {
        return FaithLabel::Polytheism(PolytheismSub::NatureForces);  // 自然力量型
    }
    if theology.deity_count > 2.0 {
        return FaithLabel::Polytheism(PolytheismSub::Functional);  // 职能分工型多神教
    }
    if theology.deity_count >= 0.8 && theology.deity_count <= 1.2 {
        if theology.exclusivity > 0.7 { return FaithLabel::Monotheism(MonotheismSub::Exclusive); }
        if theology.mysticism > 0.6 { return FaithLabel::Monotheism(MonotheismSub::Mystical); }
        return FaithLabel::Monotheism(MonotheismSub::Standard);
    }
    if theology.deity_count < 0.5 {
        if theology.mysticism > 0.5 && theology.hierarchy_degree < 0.3 {
            return FaithLabel::Shamanism;
        }
        return FaithLabel::Animism;
    }
    FaithLabel::Other  // 不属于常见分类——唯一的信仰
}
```

**FaithLabel** 仅用于 UI 分类标签和 NPC 对话中的快捷引用——不影响任何系统逻辑。

## 三、FaithName——信仰命名

对标 CultureName（文化 001 §4.3）。双层体系：内名（Endonym——信徒自己的称呼）+ 外名（Exonym——外人怎么称呼）。

```rust
pub struct FaithName {
    /// 内名——音系规则生成
    pub endonym: String,
    /// 外名——5 源加权随机
    pub exonyms: Vec<ExonymEntry>,
}

pub struct ExonymEntry {
    pub name: String,
    pub source: ExonymSource,
    pub weight: f32,
}

pub enum ExonymSource {
    /// 从神名派生（"宙斯教"）——多神信仰最常见的命名方式
    FromDeity { deity_name: String },
    /// 从起源地命名（"麦加教"、"耶路撒冷教"）
    FromPlace { place_name: String },
    /// 从创始人命名（"基督教"→ Christ·基督）
    FromFounder { founder_name: String },
    /// 从核心教义命名（"伊斯兰"→ 顺从/和平）
    FromDoctrine { doctrine_keyword: String },
    /// 从实践命名（"浸信会"→ Baptism·浸礼）
    FromPractice { practice_name: String },
}
```

**生成规则**：
- endonym 从信仰所在 origin_culture 的音系规则生成（复用语言表达系统 002 的音系生成器）
- exonyms 生成 2-5 个，每个从不同 source + 不同邻近文化的视角生成
- UI 显示：优先使用 endonym；如果玩家信仰不同 → 显示 exonym（"你们叫X教，我们叫Y教"）

## 四、FaithCalendarQuery 相关类型定义

这些类型被 `FaithCalendarQuery` trait（由 [[../文化系统/008-节日与仪式系统|文化 008]] 定义、本模块实现）引用——此前为前向引用，现完整定义。

### 4.1 HolyDay——神圣日

```rust
/// 信仰的一个神圣日——供 FaithCalendarQuery::holy_days() 返回。
pub struct HolyDay {
    /// 节日名称："冬至祭天"、"复活节"、"斋月"
    pub name: String,
    /// 日期指定——使用游戏 120 天/年历法
    pub date: HolyDayDate,
    /// 遵守方式
    pub observance: ObservanceType,
    /// 在信仰中的重要性 (0.0-1.0)
    pub importance: f32,
    /// 关联的神祇——如为此神的节日则为 Some
    pub associated_deity: Option<String>,
}

/// 日期指定——适应游戏 120 天/年历法。
/// 使用 Seasonal/Computed 变体而非硬编码 day_of_year——
/// 使得模组可以改变历年长度而不破坏节日逻辑。
pub enum HolyDayDate {
    /// 固定日——每年第 N 天 (0-119)
    FixedDay(u16),
    /// 季节事件——不依赖特定天数
    Seasonal { event: SeasonalHolyDayEvent, offset_days: i8 },
    /// 阴历——月相
    Lunar { phase: u8, month: u8 },
    /// 计算规则——简单算法（如"春分后第一个满月"）
    Computed { rule: ComputedDateRule },
}

/// 季节事件锚点——对标 120 天/年历法。
/// SpringEquinox = day 15, SummerSolstice = day 45,
/// AutumnEquinox = day 75, WinterSolstice = day 105。
pub enum SeasonalHolyDayEvent {
    SpringEquinox, SummerSolstice,
    AutumnEquinox, WinterSolstice,
    FirstHarvest,   // 第一次收获——依赖农业系统
    LastHarvest,    // 最后一次收获
    FirstSnow,      // 初雪——依赖天气系统
    FirstBloom,     // 初花
}

pub enum ComputedDateRule {
    FirstFullMoonAfterSpringEquinox,       // 春分后第一个满月——对标复活节
    NthDayAfterWinterSolstice(u16),        // 冬至后第 N 天
    FullMoonOfMonth(u8),                   // 第 N 月的满月
}

pub enum ObservanceType {
    Feast,       // 欢庆/宴饮
    Fast,        // 斋戒
    Meditation,  // 静思/禅修
    Pilgrimage,  // 朝圣
    Vigil,       // 守夜
    Atonement,   // 赎罪
    Procession,  // 游行
}
```

```rust
/// 信仰的一条斋戒规则——供 FaithCalendarQuery::fasting_rules() 返回。
```

## 五、SacredArchitectureParams——神圣建筑参数

### 5.1 数据结构

**信仰→世界生成建筑系统（P5-P6）消费。** 从 FaithTheology 派生。

```rust
/// 信仰系统定义。世界生成建筑系统消费。
pub struct SacredArchitectureParams {
    /// 神圣几何——建筑的基本空间组织
    pub geometry: SacredGeometry,
    /// 方向——建筑主轴的朝向
    pub orientation: SacredOrientation,
    /// 高度偏好
    pub height_profile: HeightProfile,
    /// 光线品质
    pub light_quality: LightQuality,
    /// 图像/象征——神像策略
    pub iconography: IconographyApproach,
    /// 材质偏好
    pub material_palette: Vec<MaterialPreference>,
    /// 空间布局类型
    pub spatial_layout: TempleLayout,
}

pub enum SacredGeometry {
    Axial，          // 轴向——沿中轴线排列（哥特式/希腊式）
    Centripetal，    // 向心——中心圣坛+环绕空间（万神殿/佛塔）
    Organic，        // 有机——顺应自然地形（神道神社/德鲁伊圣地）
    Directionless，  // 无特定方向——功能主义（贵格会聚会所）
}

pub enum SacredOrientation {
    East，           // 朝东（日出——基督教/密特拉传统）
    South，          // 朝南
    West，           // 朝西
    North，          // 朝北
    TowardHolyCity， // 朝向圣城（需要 HolyCity 坐标）
    TowardHighestPeak， // 朝向最高山峰（山岳崇拜）
    None，           // 无要求
}

pub enum HeightProfile {
    Towering，       // 高耸入云（哥特式——"离神更近"）
    Broad，          // 宽阔展开（佛寺——"容纳众生"）
    Subterranean，   // 地下（密特拉/早期基督教墓穴——"隐秘的圣地"）
    Humble，         // 朴素低矮（贵格会——"神在朴素中"）
    Adaptive，       // 随当地建筑风格——无独立偏好
}

pub enum LightQuality {
    Dark，           // 暗——烛光为主（中世纪教堂）
    Bright，         // 亮——大量天窗/开窗（佛寺/伊斯兰）
    Colored，        // 彩色——彩色玻璃（哥特式）
    Natural，        // 自然光——露天/半露天（神道/泛灵圣地）
}

pub enum IconographyApproach {
    Iconic，         // 具象——神像/圣人像/壁画叙事
    Symbolic，       // 符号——几何图形/文字/抽象符号
    Aniconic，       // 禁止偶像——空的空间（伊斯兰/早期佛教/某些新教）
    NaturalElement， // 自然元素——圣树/圣石/泉水（泛灵/神道）
}

pub enum TempleLayout {
    NaveAndAisles，      // 中殿+侧廊（教堂标准）
    Courtyard，          // 方形庭院+回廊（清真寺/佛寺标准）
    Circular，           // 圆形/同心圆（万神殿/佛塔）
    Grotto，             // 岩窟/洞穴（自然圣地）
    MultiBuilding，      // 多栋建筑群（寺院/修道院复合体）
}

pub struct MaterialPreference {
    pub material: String,       // "white_marble"/"dark_basalt"/"cedar_wood"/"gold_leaf"
    pub usage: MaterialUsage,   // 结构/装饰/地面/祭坛
    pub preference: f32,        // 0-1 偏好强度
}

pub enum MaterialUsage { Structure, Decoration, Floor, Altar, Roof }
```

### 5.2 派生规则——FaithTheology → SacredArchitectureParams

| FaithTheology 条件 | 建筑影响 |
|-------------------|---------|
| hierarchy_degree > 0.7 | Axial + Towering（强调等级——神职在远端，信众在近端） |
| mysticism > 0.7 | Dark + Symbolic/Aniconic（暗空间+抽象符号强化神秘感） |
| nature_sacredness > 0.7 | Organic + NaturalElement（顺应自然+圣树圣石） |
| ritual_formality > 0.7 | Axial + 特定方向 |
| scripture_centrality > 0.7 | Bright（需要光线阅读经典） |
| exclusivity < 0.3 | Adaptive——随当地建筑风格 |
| exclusivity > 0.8 | Iconic——具象神像强化"唯一真神"的视觉存在 |
| orthodoxy_vs_orthopraxy > 0.7 | 重点在 altar/pulpit（讲道台——"宣讲正确信仰"） |
| orthodoxy_vs_orthopraxy < 0.3 | 重点在 ritual space（仪式空间——"执行正确仪式"） |

```rust
/// 信仰对其他信仰的基本态度——存储在 FaithDefinition 中。
```

### 6.4 proselytizing()——派生方法

```rust
/// FaithTheology 的派生方法——不由字段直接存储。
/// 在 004 §5.3 和 005 §2.1 中使用。
impl FaithTheology {
    /// 传教倾向——从 exclusivity + orthodoxy_vs_orthopraxy + faith_as_identity 派生。
    /// 高排他性 + 高正信 = "你必须信对的" → 强传教
    /// 低排他性 + 低正信 = "你爱信啥信啥" → 弱传教
    pub fn proselytizing(&self) -> f32 {
        (self.exclusivity * 0.5 + self.orthodoxy_vs_orthopraxy * 0.3 + self.faith_as_identity * 0.2)
            .clamp(0.0, 1.0)
    }
}
```

### 7.2 加载规则

- 模组 TOML 在种子生成前加载
- `[[faiths]]` → 直接注入 FaithDefinition 池（世界生成时作为可选项参与分配）
- `[[faith_overrides]]` → 在种子生成的池中查找并部分覆盖参数
- 加载失败 → 跳过该条目 + 记录日志警告（不影响种子生成）
- 模组不能删除种子生成的信仰（只能修改参数或新增）

## 八、种子生成（P2.5）

### 5.1 生成管线

```
世界种子
  → 文化种子(CultureCoreParams)
  → 对每个文化:
       religiosity > 0.1 → 生成 0-3 个信仰
       religiosity 越高 → 信仰数量越多、theology 参数越极端
       → FaithTheory 参数 = seed_random(baseline) + culture_bias(clamp ±0.05)
       → 生成 FaithDefinition 全部字段
       → 注册到 FaithGenealogy
```

### 5.2 生成数量公式

```rust
/// P2.5 中每文化生成的信仰数量。
fn faith_count_for_culture(religiosity: f32, rng: &mut impl Rng) -> u32 {
    // religiosity < 0.1 → 0 信仰（完全世俗化文化）
    // religiosity = 0.5 → 1-2 信仰
    // religiosity = 1.0 → 2-3 信仰
    let base = (religiosity * 2.5) as u32;  // 0 ~ 2
    let extra = if rng.gen::<f32>() < (religiosity - 0.5).max(0.0) * 2.0 { 1 } else { 0 };
    base + extra  // 0 ~ 3
}
```

### 5.3 FaithTheology 生成

```rust
/// 从世界种子 + 文化偏移生成 FaithTheology。
fn generate_theology(culture: &CultureCoreParams, base_rng: &mut impl Rng) -> FaithTheology {
    // 独立 roll 每个参数
    let mut t = FaithTheology {
        deity_count: base_rng.gen_range(0.0..15.0),
        ancestor_importance: base_rng.gen_range(0.0..1.0),
        nature_sacredness: base_rng.gen_range(0.0..1.0),
        hierarchy_degree: base_rng.gen_range(0.0..1.0),
        scripture_centrality: base_rng.gen_range(0.0..1.0),
        ritual_formality: base_rng.gen_range(0.0..1.0),
        exclusivity: base_rng.gen_range(0.0..1.0),
        mysticism: base_rng.gen_range(0.0..1.0),
        orthodoxy_vs_orthopraxy: base_rng.gen_range(0.0..1.0),
        faith_as_identity: base_rng.gen_range(0.0..1.0),
    };

    // 文化温和偏移（上限 ±0.05）——对标 CHG-024 的群系对文化参数偏移设计
    t.ancestor_importance = (t.ancestor_importance + culture.collectivism * 0.05).clamp(0.0, 1.0);
    t.hierarchy_degree = (t.hierarchy_degree + culture.power_distance * 0.05).clamp(0.0, 1.0);
    t.ritual_formality = (t.ritual_formality + culture.uncertainty_avoidance * 0.05).clamp(0.0, 1.0);
    t.exclusivity = (t.exclusivity - culture.openness_to_outsiders * 0.05).clamp(0.0, 1.0);
    t.orthodoxy_vs_orthopraxy = (t.orthodoxy_vs_orthopraxy + culture.uncertainty_avoidance * 0.05).clamp(0.0, 1.0);
    t.faith_as_identity = (t.faith_as_identity + culture.collectivism * 0.05).clamp(0.0, 1.0);

    t
}
```

### 5.4 派生生成

```rust
/// 生成所有派生字段——基于 theology 参数。
fn derive_fields(def: &mut FaithDefinition, culture_rng: &mut impl Rng) {
    // 1. 神祇生成
    def.deities = generate_deities(def.theology.deity_count, def.theology.nature_sacredness, culture_rng);
    
    // 2. 教义生成——数量 = 3 + theology.scripture_centrality * 5
    //    doctrine 类别分布受 theology 影响
    def.doctrines = generate_doctrines(def.theology, culture_rng);
    
    // 3. 禁忌生成——数量 = theology.ritual_formality * 6
    def.taboos = generate_taboos(def.theology, culture_rng);
    
    // 4. 神圣日生成——数量 = 2 + theology.ritual_formality * 5
    def.holy_days = generate_holy_days(def.theology, culture_rng);
    
    // 5. 斋戒规则
    def.fasting_rules = generate_fasting_rules(def.theology, &def.holy_days, culture_rng);
    
    // 6. 神职角色
    def.clergy_roles = generate_clergy_roles(def.theology, culture_rng);
    
    // 7. 神圣建筑参数
    def.sacred_architecture = derive_sacred_architecture(def.theology);
    
    // 8. 葬仪
    def.funeral_customs = derive_funeral_customs(def.theology, def.doctrines, culture_rng);
    
    // 9. 命名
    def.name = generate_faith_name(&def.deities, def.theology, def.origin_culture, culture_rng);
}
```

```rust
/// 宗教机构的组织形式。
```
