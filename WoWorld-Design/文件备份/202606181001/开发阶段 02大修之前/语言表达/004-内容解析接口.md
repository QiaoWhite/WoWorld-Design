## 一、ContentResolver trait——内容解析

### 1.1 核心 trait

```rust
/// 内容解析器——每个拥有"可表达内容"的模块实现此 trait
/// lang_expression 通过此 trait 获取内容，不反向依赖任何模块
pub trait ContentResolver: Send + Sync {
    /// 该 resolver 处理哪种载体类型
    fn carrier_type(&self) -> u8;

    /// 给定模块内 ID 和读者上下文，返回可用于生成的内容
    /// 返回 None 表示: ID 不存在 / 内容已完全损毁 / 读者无权访问
    fn resolve(
        &self,
        local_id: u64,
        ctx: &ReaderContext,
    ) -> Option<ResolvedContent>;
}

/// 中间产物——包含 TextGenerator 所需的全部信息
pub struct ResolvedContent {
    pub language: LanguageId,
    pub template_id: u32,
    pub params: HashMap<String, String>,  // 键必须遵循标准参数名约定
    pub encoding: Encoding,
    pub condition: f32,                   // 载体物理状态 0.0-1.0
    pub metadata: ContentMetadata,
}

impl ResolvedContent {
    /// 从内容参数中导出可靠性提示
    pub fn reliability(&self) -> ReliabilityHint {
        // 默认由各 resolver 在构建时传入——此方法为 fallback
        ReliabilityHint::Unknown
    }
}
```

### 1.2 LocatableReadable——位置查询扩展

```rust
/// 仅对"有物理位置的可读物"实现
/// 对话记录、口头传说等没有物理位置——不实现此 trait
pub trait LocatableReadable: ContentResolver {
    /// 返回指定位置和半径内的所有可读物 ExpressionRef
    fn query_at_location(
        &self,
        location: LocationId,
        radius_m: f32,
    ) -> Vec<ExpressionRef>;

    /// 按位置和载体类型过滤
    fn query_at_location_filtered(
        &self,
        location: LocationId,
        radius_m: f32,
        carrier_types: &[u8],
    ) -> Vec<ExpressionRef>;
}
```

## 二、ExpressionRegistry——注册表

### 2.1 定义

```rust
pub struct ExpressionRegistry {
    /// carrier_type → resolver
    resolvers: HashMap<u8, Box<dyn ContentResolver>>,
    /// 文本生成器——注册表持有，所有 read() 调用共用
    generator: TextGenerator,
}

impl ExpressionRegistry {
    /// 注册一个 resolver（启动时各模块调用）
    pub fn register(&mut self, resolver: Box<dyn ContentResolver>) {
        let ct = resolver.carrier_type();
        if self.resolvers.contains_key(&ct) {
            panic!("Duplicate ContentResolver for carrier_type 0x{:02X}", ct);
        }
        self.resolvers.insert(ct, resolver);
    }

    /// 统一读取入口——O(1) 哈希分发
    pub fn read(
        &self,
        expr_ref: ExpressionRef,
        ctx: &ReaderContext,
    ) -> Option<ReadResult> {
        let resolver = self.resolvers.get(&expr_ref.carrier_type())?;
        let content = resolver.resolve(expr_ref.local_id(), ctx)?;
        Some(self.generator.generate(content, ctx))
    }

    /// 查询位置附近的所有可读物（聚合所有 LocatableReadable resolver）
    pub fn query_location(
        &self,
        location: LocationId,
        radius_m: f32,
    ) -> Vec<(ExpressionRef, ContentMetadata)> {
        self.resolvers.values()
            .filter_map(|r| {
                let locatable = (r as &dyn Any).downcast_ref::<&dyn LocatableReadable>()?;
                // 实际上需要用 trait 对象的 downcast——这里展示逻辑
                Some(locatable.query_at_location(location, radius_m))
            })
            .flatten()
            .filter_map(|expr_ref| {
                let content = self.read(expr_ref, &ReaderContext::preview());
                Some((expr_ref, content?.metadata))
            })
            .collect()
    }
}
```

### 2.2 注册表不做什么

- 不管理内容的生命周期——内容由各模块自行存储（LMDB/内存）
- 不缓存解析结果——每次 `read()` 都是新鲜查询（LMDB 已有 OS 页缓存）
- 不关心内容的语义——只做"拿到参数 → 拼文本"

## 四、ReaderContext——读者上下文

### 4.1 定义

```rust
/// 读者的完整上下文——调用者在外部收集后传入
/// lang_expression 不查询任何模块来获取这些数据
pub struct ReaderContext {
    /// 读者身份——玩家或 NPC
    pub reader_id: EntityId,

    /// 读者掌握的语言及其熟练度
    pub known_languages: HashMap<LanguageId, Proficiency>,

    /// 读者是否识字
    pub literacy: bool,

    /// 魔法/物品阅读辅助
    pub reading_aids: Vec<ReadingAid>,

    /// 环境光照——0.0(全黑) → 1.0(充足光照)
    pub illumination: f32,
}

impl ReaderContext {
    /// 构建预览用的最小上下文——不需要真实读者信息
    pub fn preview() -> Self {
        Self {
            reader_id: EntityId::default(),
            known_languages: HashMap::new(),   // 预览不关心语言——显示原始内容
            literacy: true,
            reading_aids: vec![],
            illumination: 1.0,
        }
    }

    /// 获取读者对某种语言的熟练度
    pub fn language_proficiency(&self, lang: LanguageId) -> Proficiency {
        self.known_languages.get(&lang).copied().unwrap_or(Proficiency::NONE)
    }
}
```

### 4.2 ReadingAid——阅读辅助

```rust
/// 魔法/物品提供的阅读辅助
pub enum ReadingAid {
    /// 通晓语言法术——临时提升特定语言的 proficiency
    ComprehendLanguages {
        language: LanguageId,
        effective_proficiency: Proficiency,  // 法术提供的临时熟练度
        remaining_duration: f32,             // 剩余持续时间（秒）
    },
    /// 魔法透镜——提升可读性
    MagicalLens {
        condition_boost: f32,                // 直接加到 effective_condition 上
    },
    /// 翻译结界——将内容翻译到目标语言
    TranslationField {
        target_language: LanguageId,
    },
    /// 词典/语法书——缓慢但永久
    Dictionary {
        source_language: LanguageId,
        target_language: LanguageId,
        proficiency_boost: f32,              // 查词典提供的 proficiency 加成
    },
}

impl ReadingAid {
    /// 计算该辅助对 readability 的净贡献
    pub fn condition_boost(&self) -> f32 {
        match self {
            ReadingAid::MagicalLens { condition_boost } => *condition_boost,
            ReadingAid::TranslationField { .. } => 0.0,  // 翻译不改变 readability——直接替换语言
            ReadingAid::ComprehendLanguages { effective_proficiency, .. } => {
                effective_proficiency.0  // 临时 proficiency 可直接使用
            }
            ReadingAid::Dictionary { proficiency_boost, .. } => *proficiency_boost,
        }
    }
}
```

## 五、ContentMetadata——展示元数据

```rust
/// 内容的展示用元数据——随 ReadResult 一起返回
pub struct ContentMetadata {
    /// 内容的标题（书名/碑文标题/卷轴名称）
    pub title: Option<String>,
    /// 作者/书写者
    pub author: Option<String>,
    /// 日期——可展示的年号/日期字符串
    pub date: Option<String>,
    /// 原始编码类型——UI 用此字段判断是否需要额外渲染
    pub encoding: Encoding,
    /// 语言展示名——"中原通用语"
    pub language_display: String,
    /// 文字系统展示名——"苍山汉字"
    pub script_display: Option<String>,
    /// 载体描述——"3代抄本·羊皮纸"
    pub carrier_description: Option<String>,
}

pub enum Encoding {
    Text,       // 标准文本——直接显示
    Geometric,  // 法阵几何语言——Godot 应额外渲染 Formation 图
    Audio,      // 音频记录——Godot 应额外播放音频（可选）
}

/// 信息可靠性提示
pub enum ReliabilityHint {
    /// 第一手目击——NPC 亲眼所见
    Firsthand,
    /// 第二手——听别人说的，跳了 hops 次
    Secondhand { hops: u32 },
    /// 书面记录——可能有偏见，第 N 代抄本
    Written { has_bias: bool, copy_generation: u32 },
    /// 传说/民间——多手传播，高度变形
    Folklore { estimated_age_years: u32 },
    /// 阅读者本人就是信息的原始来源
    SelfOriginated,
    /// 无法确定
    Unknown,
}
```

## 六、各模块注册规范

### 6.1 实现 ContentResolver

每个模块需要实现一个（或多个）`ContentResolver`，然后在其初始化阶段注册到 `ExpressionRegistry`：

```rust
// 启动时的注册代码（在主循环/集成层）:
let mut registry = ExpressionRegistry::new(text_generator);

// 历史模块注册
registry.register(Box::new(history::PhysicalBookResolver::new(lmdb)));
registry.register(Box::new(history::LifeTraceResolver::new(lmdb)));

// 魔法模块注册
registry.register(Box::new(magic::GrimoireResolver::new(lmdb)));
registry.register(Box::new(magic::ScrollResolver::new(lmdb)));
registry.register(Box::new(magic::MagicInscriptionResolver::new(lmdb)));

// 物品模块注册
registry.register(Box::new(items::ItemInscriptionResolver::new(lmdb)));
registry.register(Box::new(items::SignResolver::new(lmdb)));
```

### 6.2 实现 LocatableReadable

仅对有物理位置的载体实现：

```rust
// PhysicalBook 有物理位置 → 实现 LocatableReadable
impl LocatableReadable for PhysicalBookResolver { ... }

// LifeTrace 有物理位置 → 实现 LocatableReadable
impl LocatableReadable for LifeTraceResolver { ... }

// Grimoire（魔导书）有物理位置 → 实现 LocatableReadable
impl LocatableReadable for GrimoireResolver { ... }

// ItemInscription（剑上铭文）→ 物品本身有位置，铭文随物品移动 → 实现 LocatableReadable
impl LocatableReadable for ItemInscriptionResolver { ... }

// MagicProjection（法师幻影留言）→ 有固定位置 → 实现 LocatableReadable
impl LocatableReadable for MagicProjectionResolver { ... }
```

## 七、ReadResult——最终产物

```rust
/// 读取操作的最终结果——直接送往 Godot 展示
pub struct ReadResult {
    /// 生成的完整文本——屏幕显示的内容
    pub text: String,

    /// 文本的语言
    pub language: LanguageId,

    /// 最终可读性——已经综合了载体状态 × 读者能力 × 辅助
    pub condition: f32,

    /// 展示元数据——用于 UI 的标题栏/作者行/来源标注
    pub metadata: ContentMetadata,

    /// 信息可靠性提示——用于 UI 的可信度标注
    pub reliability: ReliabilityHint,
}

// Godot UI 使用示例:
// ┌──────────────────────────────┐
// │ 苍山城编年史                  │ ← metadata.title
// │ 程国史官·文谦 | 程王三世十二年  │ ← metadata.author, metadata.date
// │ 中原通用语 | 3代抄本·状态良好  │ ← metadata.language_display, carrier_description
// │ ⚠ 书面记录·有作者偏见         │ ← reliability: Written { bias: true, gen: 3 }
// ├──────────────────────────────┤
// │ 苍山城编年史——程王三世1250年  │ ← text (ReadResult.text)
// │ ，北方蛮族入侵。城防军英勇抵抗  │
// │ ，击退入侵，死伤约三千人。      │
// │                              │
// │ [法阵渲染图]                   │ ← 仅当 metadata.encoding == Geometric 时显示
// │ [▶ 播放咒语音频]              │ ← 仅当 metadata.encoding == Audio 时显示
// └──────────────────────────────┘
```
