## 〇、总览

历史事件的生成不是线性时间线上随机撒点——而是**三层驱动下的因果网络涌现**。

```
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
        三层事件生成模型
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

层级 1: 趋势层 (Trend)           —— 世界参数的慢变量
  ↓ 调制所有概率                  人口增长、资源消耗、气候周期、魔力涨落、文化漂移

层级 2: 力量层 (HistoricalForce) —— 实体之间的结构性张力
  ↓ 累积 → 超过阈值 → 爆发为事件    领土扩张力、人口压力、贸易控制、宗教传播、世仇

层级 3: 事件层 (HistoricalEvent) —— 具体的历史事件
  ↓ 实际改变世界状态              战争、条约、灾变、迁徙、发现、革命

反馈环:
  事件改变世界状态 → 世界状态变化 → 趋势更新、力量重新计算 → 新的事件概率变化
```

## 一、事件数据结构

### 1.1 核心事件类型

```rust
/// 单个历史事件——event_log 中的基本单元
struct HistoricalEvent {
    // === 身份 ===
    id: EventId,                       // 唯一标识
    event_type: EventType,             // 事件类型分类
    domain: EventDomain,               // 所属领域

    // === 时空 ===
    timestamp: i32,                    // 距"现在"的年数（负数=玩家进入前，以月为单位存储）
    location: EventLocation,           // 发生位置（聚落/区域/大陆/全球）

    // === 参与者 ===
    participants: Vec<ParticipantId>,  // 参与的个人/家族/势力/政权/团体

    // === 因果结构 ===
    cause: Vec<CausalEdge>,            // 父事件（谁导致了我）
    effects: Vec<CausalEdge>,          // 子事件（我导致了谁）

    // === 强度与范围 ===
    intensity: f32,                    // 0.0-1.0，影响传播距离和嵌入深度
    affected_population_ratio: f32,    // 受影响的人口比例 (0.0-1.0)
    duration_months: u32,              // 事件持续了几个月

    // === 叙事参数（不包含叙事文本——文本由执行引擎即时生成） ===
    narrative_params: NarrativeParams, // 事件的具体参数

    // === 痕迹 ===
    traces: Vec<Trace>,                // 留下了什么物理痕迹

    // === 元数据 ===
    generation_source: GenerationSource, // 来源：预生成 / 冲击触发 / 累积触发 / 命运种子
    is_fully_lost: bool,               // 玩家进入时是否已完全无痕迹
}

/// 事件类型——按领域和性质分类
enum EventType {
    // === 政治/军事 ===
    TerritorialWar,         // 领土战争
    CivilWar,               // 内战
    Rebellion,              // 叛乱/起义
    Conquest,               // 征服/吞并
    TreatyPeace,            // 和约签订
    TreatyAlliance,         // 盟约签订
    TreatyTrade,            // 贸易协定
    DynasticChange,         // 王朝更替
    AssassinationPolitical, // 政治刺杀
    Coup,                   // 政变
    IndependenceDeclaration,// 独立宣言

    // === 经济 ===
    TradeBoom,              // 贸易繁荣
    TradeCollapse,          // 贸易崩溃
    ResourceDiscovery,      // 发现重要资源
    ResourceDepletion,      // 资源枯竭
    Famine,                 // 饥荒
    EconomicCrisis,         // 经济危机
    TechnologicalBreakthrough, // 技术突破

    // === 文化/宗教 ===
    ReligiousReformation,   // 宗教改革
    ReligiousPersecution,   // 宗教迫害
    CulturalRenaissance,    // 文化复兴
    ReligiousCrusade,       // 宗教圣战

    // === 自然 ===
    Earthquake,             // 地震
    VolcanicEruption,       // 火山喷发
    Flood,                  // 洪水
    Drought,                // 干旱
    Plague,                 // 瘟疫
    Wildfire,               // 大火
    ClimateShift,           // 气候转变

    // === 魔法 ===
    MagicDisaster,          // 魔法灾难
    MagicBreakthrough,      // 魔法突破
    AetherSurge,            // 灵元素爆发
    MagicalPlague,          // 魔法瘟疫
    MagicalCreatureRampage, // 魔法生物肆虐
    WorldTreeEvent,         // 世界树相关事件

    // === 社会/人口 ===
    MassMigration,          // 大规模迁徙

    PopulationBoom,         // 人口膨胀

    // === 探索/发现 ===
    NewLandDiscovered,      // 发现新大陆（跨洋）
    FirstCircumnavigation,  // 首次环球航行
    CulturalContact,        // 文化首次接触
}

enum EventDomain {
    Political,
    Economic,
    Cultural,
    Religious,
    Natural,
    Magical,
    Social,
    Exploration,
    Composite,              // 跨领域复合事件
}

enum EventLocation {
    Global,
    Continental(ContinentId),
    Regional(RegionId),
    Settlement(SettlementId),
    Point(LocationId, f32), // 坐标+半径km
}

enum GenerationSource {
    PreGenerated,           // 世界生成 P9 批量生成
    ImpactTrigger,          // 运行时冲击型触发器
    AccumulationTrigger,    // 运行时累积型触发器
    FateSeed { seed_id: FateSeedId },  // 命运种子触发
    EntropyTrigger,         // 熵型（纯随机）触发器 (v1.2)
}

/// 因果边——连接两个事件
struct CausalEdge {
    from: Cause,                    // 来源
    to: Effect,                     // 目标
    weight: f32,                    // 因果权重 (0.0-1.0)
    is_primary: bool,               // 是主要原因还是辅助因素
    delay_months: u32,              // 因果延迟（从因到果经过的月数）
    causal_type: CausalType,        // 因果类型
}

enum Cause {
    Event(EventId),
    Force(ForceId),                  // 力量也可以导致事件
    Trend(TrendId),                  // 趋势也可以导致事件
}

enum Effect {
    Event(EventId),
    Force(ForceId),                  // 事件可以加强/削弱力量
    Trend(TrendId),                  // 事件可以改变趋势方向
}

enum CausalType {
    Direct,         // 直接因果——A 直接导致 B
    Contributing,   // 辅助原因——A 是 B 的原因之一
    Catalyzing,     // 催化——A 加速了本就会发生的 B
    Enabling,       // 创造条件——A 为 B 创造了条件
    Protecting,     // 保护——A 阻止了 B 的发生（负面因果边）
}

/// 参与者标识——可以是多种实体
enum ParticipantId {
    Npc(NpcId),
    Family(LineageId),
    Polity(PolityId),
    Group(GroupId),
    Settlement(SettlementId),
    Culture(CultureId),
    Religion(ReligionId),
}

/// 叙事参数——事件的具体参数，不含叙事文本
/// 叙事文本由执行引擎在使用时根据参数即时生成
struct NarrativeParams {
    /// 键值对形式的参数
    /// 例：{"attacker": "冰冠王国", "defender": "七河联盟", "disputed_region": "河谷"}
    params: HashMap<String, String>,
    
    /// 定量参数
    /// 例：{"casualties": 3000, "area_affected_km2": 500, "duration_years": 3}
    quantitative: HashMap<String, f64>,
    
    /// 结果类型
    outcome: EventOutcome,
}

enum EventOutcome {
    Decisive { winner: ParticipantId, loser: ParticipantId },
    Stalemate,
    Negotiated { treaty_event_id: EventId },
    Ongoing,
    Abandoned,
    Unknown,
}
```

```rust
/// 历史因果图
/// 不是线性时间线——是有向图，包含事件之间、力量与事件之间、

}
```

## 二、趋势层

### 2.1 趋势总览

```rust
/// 趋势——世界参数的慢变量
/// 趋势调制力量检测的概率和事件的类型分布
/// 趋势本身被事件反向影响（事件可能改变趋势的方向或速度）
struct Trend {
    id: TrendId,
    trend_type: TrendType,
    
    /// 当前数值 (0.0-1.0)
    current_value: f32,
    /// 自然变化率（每年，不受事件影响时的基准值）
    natural_rate: f32,      // 可正可负
    /// 变化方向是否可被事件反转
    reversible: bool,
    
    /// 此趋势调制哪些事件类型的概率
    modulates: Vec<(EventType, f32)>,  // (事件类型, 调制系数)
    
    /// 历史事件对此趋势的改变记录
    modification_history: Vec<(EventId, f32)>,
}

enum TrendType {
    PopulationGrowth,       // 人口增长（受承载容量约束）
    ResourceDepletion,      // 资源消耗
    CulturalComplexity,     // 文化复杂化（科技/知识累积）
    ClimateCycle,           // 气候周期
    MagicEbbAndFlow,        // 魔力涨落（v1.2+）
    Secularization,         // 世俗化程度
    Urbanization,           // 城市化率
}
```

### 2.2 趋势更新

```rust
/// 每个月的 tick 中更新趋势
fn tick_trends(trends: &mut HashMap<TrendId, Trend>, world: &WorldState, dt_months: u32) {
    let dt_years = dt_months as f32 / 12.0;
    
    for trend in trends.values_mut() {
        // 基础自然变化
        let natural_delta = trend.natural_rate * dt_years;
        
        // 世界状态修正
        let state_modifier = match trend.trend_type {
            TrendType::PopulationGrowth => {
                // 人口增长率受承载容量制约
                let global_pop_ratio = world.total_population / world.total_carrying_capacity;
                if global_pop_ratio > 0.9 {
                    -0.02  // 接近上限——增长放缓
                } else if global_pop_ratio < 0.3 {
                    0.02   // 人口稀疏——增长加快
                } else {
                    0.0
                }
            }
            TrendType::ResourceDepletion => {
                // 资源消耗率受科技水平影响
                let avg_tech = world.average_tech_level();
                avg_tech * 0.01  // 科技越高——资源消耗越快
            }
            TrendType::CulturalComplexity => {
                // 文化复杂化受人口规模和聚落间连通性影响
                let connectivity = world.average_settlement_connectivity();
                connectivity * 0.005
            }
            TrendType::ClimateCycle => {
                // 气候周期是循环性的——正弦波基准
                // 在 200-500 年周期上缓慢波动
                0.0  // 基准由世界参数中的气候周期设定
            }
            _ => 0.0,
        };
        
        // 应用变化
        trend.current_value += (natural_delta + state_modifier) * dt_years;
        trend.current_value = trend.current_value.clamp(0.0, 1.0);
    }
}
```

### 2.3 v1.0 趋势层范围

v1.0 仅实现三个趋势：
- **人口增长**——用世界状态的 total_population vs total_carrying_capacity
- **资源消耗**——跟踪关键资源（铁矿/木材/粮食）的储量变化
- **文化漂移**——文化参数（honor_weight, religious_exclusivity, grudge_duration 等）的缓慢漂移

v1.1 添加魔力涨落和文化复杂化。
v1.2 添加完整的气候周期和世俗化。

## 三、力量层

### 3.1 力量的本质

**力量不是事件。力量是历史未决的张力。** 它是系统扫描世界状态后识别到的"结构性冲突可能"。

力量不需要名字——它只是一个数据签名：(主体, 客体, 冲突类型, 当前张力)。名字是 NPC 后来给它的。不同 NPC 给它不同的名字。

```rust
/// 历史力量——两个实体之间的结构性张力
struct HistoricalForce {
    id: ForceId,
    force_type: ForceType,
    
    /// 张力的两端
    subject: ParticipantId,     // 谁在推动
    object: ParticipantId,      // 与谁对抗
    
    /// 争夺的焦点
    over: Option<RegionId>,     // 争夺什么区域/资源
    
    /// 张力数值
    intensity: f32,             // 当前张力 (0.0-1.0)
    buildup_rate: f32,          // 自然累积速度（每年）
    
    /// 生命周期
    active_since: i32,          // 从何时开始存在
    last_erupted: Option<i32>,  // 最近一次爆发为事件的时间
    dissolved_at: Option<i32>,  // 若已消解——何时消解
    
    /// 是否活跃
    is_active: bool,
    
    /// 来源——是什么导致了这个力量
    source: ForceSource,
    
    /// 此力量调制哪些事件类型的概率
    /// 领土扩张力 → 战争事件概率×3 + 外交事件概率×2
    event_modulation: Vec<(EventType, f32)>,
}

enum ForceType {
    TerritorialExpansion,   // 领土扩张张力
    TradeControl,           // 贸易控制/垄断竞争 (v1.1)
    ReligiousSpread,        // 宗教传播/冲突 (v1.1)
    PopulationPressure,     // 人口压力——对邻近低密度区域的推力
    ResourceCompetition,    // 资源竞争 (v1.2)
    MagicResourceControl,   // 魔法资源控制竞争 (v1.2)
    GenerationalFeud,       // 世仇 (v1.2)
    IdeologicalConflict,    // 意识形态冲突 (v1.2)
}

enum ForceSource {
    /// 从世界状态的结构性扫描中检测到
    StructuralDetection,
    /// 从因果链中继承——以前的事件留下了未解决的冲突
    CausalInheritance { originating_events: Vec<EventId> },
    /// 从趋势累积中产生
    TrendAccumulation { driving_trend: TrendId },
}
```

```rust
/// 力量检测——每次 tick 扫描世界状态，识别和更新力量

/// 检测器 1: 领土扩张力
fn detect_territorial_forces(world: &WorldState, event_log: &HistoryGraph) -> Vec<HistoricalForce> {
    let mut forces = Vec::new();
    
    for (polity_a, polity_b) in adjacent_polity_pairs(world) {
        // 共享边界的两个政权
        let border_length = shared_border_length(polity_a, polity_b, world);
        if border_length < 10.0 {  // 边界太短——排除
            continue;
        }
        
        // 资源梯度——相邻区域资源差异越大，扩张动机越强
        let resource_a = polity_total_resources(polity_a, world);
        let resource_b = polity_total_resources(polity_b, world);
        let resource_gradient = ((resource_b - resource_a).abs() / resource_a.max(resource_b)).min(1.0);
        
        // 历史冲突频次——以前打过多少次
        let past_conflicts = count_conflicts_between(polity_a, polity_b, event_log);
        let historical_hostility = (1.0 - (-0.5 * past_conflicts as f32).exp()).min(1.0);
        
        // 文化差异
        let cultural_distance = 1.0 - culture_similarity(
            polity_a.dominant_culture(world),
            polity_b.dominant_culture(world),
        );
        
        // 计算张力
        let tension = border_length / 500.0 * 0.15
                    + resource_gradient * 0.35
                    + historical_hostility * 0.35
                    + cultural_distance * 0.15;
        
        if tension > 0.2 {  // 最低阈值——只记录值得关注的张力
            let disputed = shared_border_region(polity_a, polity_b, world);
            
            // 确定扩张方向——资源较少的一方更可能想扩张
            let (subject, object) = if resource_a < resource_b {
                (polity_a.into(), polity_b.into())
            } else {
                (polity_b.into(), polity_a.into())
            };
            
            forces.push(HistoricalForce {
                id: ForceId::new(),
                force_type: TerritorialExpansion,
                subject,
                object,
                over: disputed,
                intensity: tension.min(1.0),
                buildup_rate: tension * 0.08,  // 高张力 → 累积更快（正反馈）
                active_since: last_conflict_between(polity_a, polity_b, event_log)
                    .map(|e| e.timestamp)
                    .unwrap_or(timestamp),  // 从上次冲突开始算
                last_erupted: last_conflict_between(polity_a, polity_b, event_log)
                    .map(|e| e.timestamp),
                dissolved_at: None,
                is_active: true,
                source: ForceSource::StructuralDetection,
                event_modulation: vec![
                    (TerritorialWar, 3.0),
                    (TreatyAlliance, -0.5),     // 高张力降低联盟概率
                    (TreatyPeace, 2.0),          // 但也增加和谈动机
                ],
            });
        }
    }
    
    forces
}

/// 检测器 2: 人口压力力
fn detect_population_pressure(world: &WorldState) -> Vec<HistoricalForce> {
    let mut forces = Vec::new();
    
    for polity in world.polities.iter() {
        let pop_ratio = polity.population as f32 / polity.carrying_capacity as f32;
        
        if pop_ratio > 0.85 {
            // 人口接近承载上限 → 对所有邻接的低密度政权产生压力
            for neighbor in adjacent_polities(polity, world) {
                let neighbor_ratio = neighbor.population as f32 / neighbor.carrying_capacity as f32;
                
                if neighbor_ratio < 0.5 {
                    // 邻接区域人口稀疏 → 人口扩张压力
                    let pressure = (pop_ratio - neighbor_ratio - 0.3) * 0.6;
                    
                    forces.push(HistoricalForce {
                        id: ForceId::new(),
                        force_type: PopulationPressure,
                        subject: polity.id.into(),
                        object: neighbor.id.into(),
                        over: Some(shared_border_region(polity, neighbor, world)),
                        intensity: pressure.min(1.0),
                        buildup_rate: polity.population_growth_rate * 0.5,
                        active_since: timestamp_at(pop_ratio > 0.8, world),
                        last_erupted: None,
                        dissolved_at: None,
                        is_active: true,
                        source: ForceSource::TrendAccumulation {
                            driving_trend: trend_id(PopulationGrowth),
                        },
                        event_modulation: vec![
                            (MassMigration, 2.5),
                            (TerritorialWar, 2.0),
                            (Famine, 1.5),
                        ],
                    });
                }
            }
        }
    }
    
    forces
```

### 3.3 力量的更新与消解

```rust
/// 每个 tick 更新已有的力量
fn tick_forces(
    forces: &mut Vec<HistoricalForce>,
    recent_events: &[HistoricalEvent],
    world: &WorldState,
    dt_months: u32,
) {
    let dt_years = dt_months as f32 / 12.0;
    
    for force in forces.iter_mut() {
        // 自然累积——未解决的张力随时间增长
        force.intensity += force.buildup_rate * dt_years;
        force.intensity = force.intensity.min(1.0);
        
        // 检查是否有最近的事件缓解或加剧了张力
        for event in recent_events {
            // 如果一次外交事件缓解了张力
            if event_involves(event, force.subject) && event_involves(event, force.object) {
                match event.event_type {
                    TreatyPeace | TreatyAlliance => {
                        force.intensity *= 0.6;
                        force.buildup_rate *= 0.5;
                    }
                    TreatyTrade => {
                        force.intensity *= 0.85;
                    }
                    TerritorialWar | Conquest => {
                        // 如果爆发为战争——看结果
                        match &event.narrative_params.outcome {
                            EventOutcome::Decisive { winner, loser } => {
                                if *winner == force.subject {
                                    // 主体胜利——张力大幅下降（目的达到了）
                                    force.intensity *= 0.2;
                                } else {
                                    // 主体战败——张力可能反升（"要报仇"）
                                    force.intensity *= 1.2;
                                    force.buildup_rate *= 1.3;
                                }
                            }
                            EventOutcome::Negotiated { .. } => {
                                force.intensity *= 0.5;
                            }
                            _ => {}
                        }
                    }
                    _ => {}
                }
            }
        }
        
        // 结构消解条件
        // 如果一方已不存在
        if !entity_exists(force.subject, world) || !entity_exists(force.object, world) {
            force.dissolved_at = Some(world.current_timestamp);
            force.is_active = false;
            force.intensity *= 0.1;
        }
        
        // 如果强度过低——自然消解
        if force.intensity < 0.08 {
            force.dissolved_at = Some(world.current_timestamp);
            force.is_active = false;
        }
    }
    
    // 移除已消解且不再相关的力量（保留最近消解的用于历史记录）
    forces.retain(|f| {
        f.is_active || 
        f.dissolved_at.map_or(true, |d| world.current_timestamp - d < 120)  // 保留近 10 年
    });
}
```

### 4.2 因果传播引擎

```rust
/// 因果传播——从已有事件递归生成子事件
struct CausalPropagationEngine {
    config: PropagationConfig,
    graph: HistoryGraph,
    rng: StdRng,
    // 世界状态快照——传播过程中会修改
    world_snapshot: WorldState,
}

impl CausalPropagationEngine {
    /// 主传播循环——从根事件开始
    fn propagate(
        &mut self,
        root_events: Vec<HistoricalEvent>,
        start_timestamp: i32,
        end_timestamp: i32,
    ) -> HistoryGraph {
        // 事件队列——按时间顺序处理
        let mut event_queue: BinaryHeap<EventQueueEntry> = BinaryHeap::new();
        
        for root in root_events {
            event_queue.push(EventQueueEntry {
                event: root,
                recursion_depth: 0,
                priority: root.intensity,
            });
        }
        
        let total_ticks = (end_timestamp - start_timestamp) / 1;  // 1个月步长
        
        for tick in 0..total_ticks {
            let current_timestamp = start_timestamp + tick;
            
            // 处理当前 tick 的队列中事件
            while let Some(entry) = event_queue.peek() {
                if entry.event.timestamp > current_timestamp {
                    break;
                }
                let entry = event_queue.pop().unwrap();
                
                // 添加事件到图中
                self.graph.events.insert(entry.event.id, entry.event.clone());
                self.update_indices(&entry.event);
                
                // 更新世界状态快照
                self.apply_event_to_world(&entry.event);
                
                // 生成子事件
                let children = self.generate_children(&entry);
                for child in children {
                    if self.graph.events.len() < self.config.max_total_events as usize {
                        event_queue.push(child);
                    }
                }
            }
            
            // 趋势更新
            self.tick_trends(1);
            
            // 力量检测
            let forces = self.detect_forces(current_timestamp);
            for force in &forces {
                self.graph.forces.insert(force.id, force.clone());
                
                // 检查力量是否触发事件
                if force.intensity > self.config.trigger_thresholds.force_to_event {
                    if let Some(event) = self.force_to_event(force, current_timestamp) {
                        event_queue.push(EventQueueEntry {
                            event,
                            recursion_depth: 0,
                            priority: force.intensity,
                        });
                    }
                }
            }
            
            // 力量更新
            self.tick_forces(1);
        }
        
        self.graph.clone()
    }
    
    /// 为给定事件生成子事件
    fn generate_children(&mut self, entry: &EventQueueEntry) -> Vec<EventQueueEntry> {
        if entry.recursion_depth >= self.config.max_recursion_depth {
            return vec![];
        }
        
        let event = &entry.event;
        let template = get_causal_template(event.event_type);
        let mut children = Vec::new();
        
        for potential_effect in &template.possible_effects {
            // 基础概率 × 事件强度 × 世界上下文 × 趋势调制 × 力量调制
            let mut prob = potential_effect.base_probability;
            
            // 递归衰减
            prob *= self.config.probability_decay_per_level.powi(entry.recursion_depth as i32);
            
            // 事件强度调制
            prob *= event.intensity;
            
            // 世界上下文调制
            prob *= self.world_context_modifier(&potential_effect.effect_type, event);
            
            // 趋势调制
            for (trend_id, trend) in &self.graph.trends {
                if let Some(modifier) = trend.modulates.iter().find(|(t, _)| *t == potential_effect.effect_type) {
                    prob *= 1.0 + modifier.1 * trend.current_value;
                }
            }
            
            // 力量调制
            for force in self.graph.forces.values() {
                if force.is_active && force_involves(force, &event.participants) {
                    for (event_type, modifier) in &force.event_modulation {
                        if *event_type == potential_effect.effect_type {
                            prob *= 1.0 + *modifier * force.intensity;
                        }
                    }
                }
            }
            
            // 条件检查
            if let Some(condition) = &potential_effect.condition {
                if !condition.check(event, &self.world_snapshot) {
                    continue;  // 条件不满足——跳过
                }
            }
            
            // 掷骰子
            if self.rng.gen::<f32>() < prob {
                let delay_months = self.rng.gen_range(potential_effect.delay_range.clone());
                let child_timestamp = event.timestamp + delay_months as i32;
                
                let child = HistoricalEvent {
                    id: EventId::new(),
                    event_type: potential_effect.effect_type,
                    domain: domain_of(potential_effect.effect_type),
                    timestamp: child_timestamp,
                    location: self.derive_child_location(event, &potential_effect.effect_type),
                    participants: self.derive_child_participants(event, &potential_effect.effect_type),
                    cause: vec![CausalEdge {
                        from: Cause::Event(event.id),
                        to: Effect::Event(EventId::PLACEHOLDER),  // 会在插入时修正
                        weight: prob,
                        is_primary: prob > 0.5,
                        delay_months,
                        causal_type: potential_effect.causal_type,
                    }],
                    effects: vec![],
                    intensity: event.intensity * prob * 0.8 + 0.2,
                    affected_population_ratio: potential_effect.typical_population_impact,
                    duration_months: typical_duration(potential_effect.effect_type, &mut self.rng),
                    narrative_params: self.derive_child_params(event, &potential_effect.effect_type),
                    traces: vec![],
                    generation_source: PreGenerated,
                    is_fully_lost: false,
                };
                
                children.push(EventQueueEntry {
                    event: child,
                    recursion_depth: entry.recursion_depth + 1,
                    priority: entry.priority * 0.7,
                });
            }
        }
        
        children
    }
    
    /// 世界上下文调制——事件类型的概率受当前世界状态影响
    fn world_context_modifier(&self, event_type: &EventType, parent: &HistoricalEvent) -> f32 {
        match event_type {
            Famine => {
                // 如果区域粮食产量已经很低 → 饥荒概率↑
                let food_sufficiency = self.world_snapshot.food_sufficiency_at(&parent.location);
                if food_sufficiency < 0.5 { 2.0 } else { 1.0 }
            }
            Plague => {
                // 如果区域人口密度高 → 瘟疫概率↑
                let pop_density = self.world_snapshot.population_density_at(&parent.location);
                if pop_density > 0.7 { 1.8 } else if pop_density < 0.2 { 0.3 } else { 1.0 }
            }
            Rebellion => {
                // 如果政权压迫性高且人口中贫困比例高 → 叛乱概率↑
                let polity = self.world_snapshot.polity_at(&parent.location);
                let oppression = polity.map(|p| p.oppression_index).unwrap_or(0.5);
                let poverty = self.world_snapshot.poverty_rate_at(&parent.location);
                oppression * poverty * 3.0
            }
            MassMigration => {
                // 如果邻近区域有更好的生存条件 → 迁徙概率↑
                1.0  // 简化——更精确需要邻近区域对比
            }
            _ => 1.0,  // 默认无调制
        }
    }
}
```

## 五、命运种子

### 5.1 定义与定位

命运种子是**基于世界状态结构性张力的、可能触发也可能永不触发的未来事件蓝图**。

关键原则：
- **命运种子不直接影响任何 NPC 的行为**。它不是"命运"，只是"高度可能的结构性结果"
- **不含任何叙事内容**。没有预言文本、没有预兆、没有梦境
- **纯化为结构规则**——触发条件完全是世界参数的数值组合
- **可能永远不触发**——如果条件不满足或条件在触发前消失

### 5.2 数据结构

### 6.2 冲击型触发器（v1.0）

```rust
struct ImpactTriggerSystem {
    triggers: Vec<HistoryTrigger>,
}

impl ImpactTriggerSystem {
    /// 每月 tick——检查世界状态变更
    fn tick(&mut self, world_snapshot: &WorldState, previous_snapshot: &WorldState, current_timestamp: i32) -> Vec<HistoricalEvent> {
        let mut triggered_events = Vec::new();
        
        // 计算世界状态变更
        let delta = WorldStateDelta::compute(previous_snapshot, world_snapshot);
        
        for trigger in &mut self.triggers {
            if let HistoryTrigger::Impact { metric, threshold, direction, cooldown_years, last_triggered } = trigger {
                // 冷却检查
                if let (Some(cooldown), Some(last)) = (cooldown_years, last_triggered) {
                    if current_timestamp - *last < *cooldown * 12 {
                        continue;  // 冷却中——跳过
                    }
                }
                
                // 评估当前指标值
                let current_value = world_snapshot.evaluate_metric(metric);
                
                let should_trigger = match direction {
                    ThresholdDirection::Above => current_value > *threshold,
                    ThresholdDirection::Below => current_value < *threshold,
                };
                
                if should_trigger {
                    // 创建事件
                    let event = self.create_impact_event(metric, current_value, &delta, current_timestamp);
                    
                    // 因果链豁免——如果此事件是已有事件的直接后果，降低阈值
                    // (由于函数较长，此处省略因果链检查的详细实现)
                    
                    triggered_events.push(event);
                    *last_triggered = Some(current_timestamp);
                }
            }
        }
        
        triggered_events
    }
    
    fn create_impact_event(&self, metric: &WorldMetric, value: f32, delta: &WorldStateDelta, timestamp: i32) -> HistoricalEvent {
        let (event_type, location, intensity) = match metric {
            WorldMetric::NpcDeathRate { region } => {
                if value > 0.15 {
                    (Plague, EventLocation::Regional(*region), (value * 5.0).min(0.95))
                } else {
                    (Famine, EventLocation::Regional(*region), value * 3.0)
                }
            }
            WorldMetric::BuildingDestructionRate { region } => {
                if delta.caused_by_magic {
                    (MagicDisaster, EventLocation::Regional(*region), value * 4.0)
                } else {
                    (Earthquake, EventLocation::Regional(*region), value * 3.0)
                }
            }
            WorldMetric::PopulationChange { region } => {
                if value < -0.3 {
                    (PopulationCollapse, EventLocation::Regional(*region), (-value).min(1.0))
                } else if value > 0.5 {
                    (PopulationBoom, EventLocation::Regional(*region), value)
                } else {
                    (MassMigration, EventLocation::Regional(*region), 0.5)
                }
            }
            WorldMetric::ResourceDepletionRate { resource, region } => {
                (ResourceDepletion, EventLocation::Regional(*region), value * 2.0)
            }
            WorldMetric::MagicInstabilityIndex { region } => {
                (MagicDisaster, EventLocation::Regional(*region), value * 3.0)
            }
            _ => (PopulationCollapse, EventLocation::Global, 0.5),  // fallback
        };
        
        HistoricalEvent {
            id: EventId::new(),
            event_type,
            domain: domain_of(event_type),
            timestamp,
            location,
            participants: vec![],
            cause: vec![],
            effects: vec![],
            intensity,
            affected_population_ratio: 0.1,
            duration_months: typical_duration(event_type, &mut StdRng::from_entropy()),
            narrative_params: NarrativeParams {
                params: HashMap::new(),
                quantitative: HashMap::new(),
                outcome: EventOutcome::Ongoing,
            },
            traces: vec![],
            generation_source: GenerationSource::ImpactTrigger,
            is_fully_lost: false,
        }
    }
}
```

### 6.3 阈值策略总结

| 事件级别 | impact_score | 策略 |
|---------|-------------|------|
| 全球级 | > 0.7 | 无条件记录，无冷却 |
| 区域级 | 0.4-0.7 | 固定阈值，同区域冷却 2-5 年 |
| 本地级 | 0.2-0.4 | 不进 event_log，走 NPC 个体记忆 |
| 因果链事件 | — | 降低阈值 0.15（保证因果链完整） |
| 累积触发 | — | 自己的条件自己判断（不影响冲击触发配额） |

```rust
fn apply_event_to_world(event: &HistoricalEvent, world: &mut WorldState) -> Vec<WorldStateDelta> {
    let mut deltas = Vec::new();
    
    match event.event_type {
        TerritorialWar | Conquest => {
            // 边界变更——根据战争结果
            if let EventOutcome::Decisive { winner, loser } = &event.narrative_params.outcome {
                if let Some(region) = disputed_region_from_event(event) {
                    let delta = WorldStateDelta::BorderShift {
                        polity_a: (*winner).as_polity().unwrap(),
                        polity_b: (*loser).as_polity().unwrap(),
                        territory: region,
                    };
                    world.apply_border_shift(&delta);
                    deltas.push(delta);
                }
            }
            
            // 人口减少
            if let Some(casualties) = event.narrative_params.quantitative.get("casualties") {
                let delta = WorldStateDelta::PopulationChange {
                    location: event.location.as_region().unwrap_or_default(),
                    delta: -(*casualties as i32),
                    cause: PopChangeCause::War,
                };
                world.apply_population_change(&delta);
                deltas.push(delta);
            }
        }
        
        Plague => {
            if let Some(pop_loss) = event.narrative_params.quantitative.get("population_loss") {
                let delta = WorldStateDelta::PopulationChange {
                    location: event.location.as_region().unwrap_or_default(),
                    delta: -(*pop_loss as i32),
                    cause: PopChangeCause::Disease,
                };
                world.apply_population_change(&delta);
                deltas.push(delta);
            }
        }
        
        Earthquake | VolcanicEruption => {
            if let Some(damage) = event.narrative_params.quantitative.get("building_damage_ratio") {
                for building in world.buildings_in(event.location.as_region().unwrap_or_default()) {
                    let delta = WorldStateDelta::BuildingDamage {
                        building_id: building.id,
                        damage_ratio: *damage,
                        cause: DamageCause::NaturalDisaster,
                    };
                    world.apply_building_damage(&delta);
                    deltas.push(delta);
                }
            }
        }
        
        SettlementAbandoned => {
            if let Some(settlement_id) = event.narrative_params.params.get("settlement_id") {
                let delta = WorldStateDelta::SettlementAbandoned {
                    settlement_id: parse_id(settlement_id),
                    reason: AbandonmentReason::Depopulation,
                };
                world.apply_settlement_abandoned(&delta);
                deltas.push(delta);
            }
        }
        
        MagicDisaster => {
            // 灵元素疤痕
            let delta = WorldStateDelta::AetherScar {
                location: event.location.as_point().unwrap_or_default().0,
                intensity: event.intensity,
                element: event.narrative_params.params.get("element")
                    .map(|e| parse_element(e))
                    .unwrap_or(Element::Ling),
            };
            world.apply_aether_scar(&delta);
            deltas.push(delta);
        }
        
        // ... 其他事件类型的处理
        _ => {}
    }
    
    deltas
}
```

```rust
/// 世界生成 P9——历史模拟
/// 此函数被世界生成管线调用
fn simulate_history(
    world: &mut WorldState,
    config: &HistorySimulation,
    seed: u64,
) -> HistoryResult {
    let start = Instant::now();
    let mut rng = StdRng::seed_from_u64(seed);
    
    // Step 1: 初始化趋势层
    let mut trends = init_trends(world, config);
    
    // Step 2: 选择根事件
    let root_events = select_root_events(world, seed);
    
    // Step 3: 初始力量检测
    let mut forces = detect_forces(world, &HistoryGraph::empty(), world.current_timestamp);
    
    // Step 4: 创建因果传播引擎
    let mut engine = CausalPropagationEngine {
        config: config.propagation_config.clone(),
        graph: HistoryGraph {
            events: HashMap::new(),
            forces: forces.iter().map(|f| (f.id, f.clone())).collect(),
            trends: trends.iter().map(|t| (t.id, t.clone())).collect(),
            edges: Vec::new(),
            root_events: root_events.iter().map(|e| e.id).collect(),
            spatial_index: BTreeMap::new(),
            participant_index: HashMap::new(),
            temporal_index: BTreeMap::new(),
            type_index: HashMap::new(),
        },
        rng: rng.clone(),
        world_snapshot: world.clone(),
    };
    
    // Step 5: 因果传播（主循环）
    let graph = engine.propagate(
        root_events,
        world.current_timestamp - config.years_to_simulate as i32 * 12,
        world.current_timestamp,
    );
    
    // Step 6: 为关键事件生成 AetherImprint
    for event in graph.events.values() {
        if event.intensity > 0.5 {
            let imprint = AetherImprint::from_event(event, world);
            world.add_aether_imprint(imprint);
        }
    }
    
    // Step 7: 更新世界状态——事件的影响已在上面的 tick 中实时应用
    
    // Step 8: 播种命运种子
    let fate_seeds = plant_fate_seeds(world, &graph, world.current_timestamp);
    
    // Step 9: 产出结果
    let elapsed = start.elapsed();
    
    HistoryResult {
        graph,
        fate_seeds,
        total_events: graph.events.len() as u32,
        generation_time_ms: elapsed.as_millis() as u32,
        years_simulated: config.years_to_simulate,
    }
}
```
