## 二、保真度数学模型

```rust
// woworld_core/src/transmission.rs

pub fn transmission_fidelity(
    source_model: &ModelTransmissionRef,
    path: TransmissionPath,
    hops: u32,
    receiver: &ReceiverFactors,
) -> f32 {
    let base = match &path {
        TransmissionPath::WrittenBook { copy_generation, physical_condition } => {
            0.95 * (0.98_f32).powi(*copy_generation as i32) * physical_condition
        }
        TransmissionPath::DeepMentorship { years, teacher_wisdom } => {
            0.70 + (years / 20.0).min(0.20) + teacher_wisdom * 0.10
        }
        TransmissionPath::Apprenticeship { years } => {
            0.55 + (years / 10.0).min(0.15)
        }
        TransmissionPath::OralTradition => 0.50,
        TransmissionPath::Observation => 0.40,
        TransmissionPath::ThoughtImprint { emotional_charge } => {
            0.15 + emotional_charge * 0.25
        }
    };

    let hop_decay = (-path.decay_rate() * hops as f32).exp();
    let receiver_factor = receiver.analytic_intuitive * 0.40
                        + receiver.flexibility * 0.30
                        + receiver.literacy * 0.30;
    let complexity_penalty = if source_model.pattern.is_complex() { 0.75 } else { 1.0 };

    (base * hop_decay * receiver_factor * complexity_penalty).clamp(0.0, 1.0)
}
```

## 三、知识重新发现

```rust
pub fn rediscover_from_reading(book: &PhysicalBook, reader: &NpcData) -> Vec<MentalModelCandidate> {
    let mut candidates = Vec::new();
    if let Some(encoding) = &book.source_concepts {
        let reader_space = effective_concept_space(reader);
        let matches = classify_pattern(encoding.pattern_sig, &reader_space);

        for (concept, match_confidence) in &matches {
            if reader.has_similar_model(&encoding.pattern_sig, 0.7) { continue; }  // 已知道

            let cultural_adjustment = if book.source_culture != Some(reader.birth_culture()) { 0.85 }
                                    else { 1.0 };

            candidates.push(MentalModelCandidate {
                pattern: encoding.to_pattern_expression(),
                initial_confidence: match_confidence * 0.6 * cultural_adjustment,
                source: ModelSource::ReadFromBook(book.work_id),
                hypothesis: true,  // 标记为假说——需要验证
            });
        }
    }
    candidates
}
```

**涌现场景**: 一本 500 年前的书可能记载了一个已被遗忘的信念。反思型 NPC 读到后能将信念重新引入世界——带较低置信度，等待验证。

```rust
/// 从已有参数派生——零新门控
pub fn knowledge_preservation_impulse(npc: &NpcData) -> Option<WritingImpulse> {
    if npc.agent_snapshot().health_penalty > 0.5 { return None; }
    if npc.agent_snapshot().literacy_best < 0.3 { return None; }

    let wisdom = wisdom_effective(npc.agent_snapshot(), &npc.mental_models, npc.age_years());
    if wisdom < WRITING_IMPULSE_MIN_WISDOM { return None; }  // [TUNING] 0.4

    let valuable: Vec<&MentalModel> = npc.mental_models.iter()
        .filter(|m| m.confidence > 0.8)
        .filter(|m| m.emotional_charge > 0.5 || m.source == ModelSource::SelfDerived)
        .collect();
    if valuable.is_empty() { return None; }

    // 紧迫感——从年龄和死亡风险派生
    let mortality_awareness = npc.agent_snapshot().gompertz_mortality_risk
                            + (npc.age_years() / 80.0).min(0.5);
    let urgency = (mortality_awareness * 0.5 + wisdom * 0.3 + (1.0 - npc.agent_snapshot().health_penalty) * 0.2)
        .clamp(0.0, 1.0);
    if urgency < WRITING_IMPULSE_MIN_URGENCY { return None; }  // [TUNING] 0.3

    Some(WritingImpulse {
        modality: SurfacingModality::Writing,
        urgency,
        target_concepts: valuable.into_iter().take(5).map(|m| m.pattern.clone()).collect(),
        purpose: WritingPurpose::KnowledgePreservation {
            intended_audience: Audience::FutureGenerations,
        },
    })
}
```

## 五、聚落知识健康度

```rust
pub fn settlement_knowledge_health(
    settlement: &SettlementId,
    books: &[PhysicalBook],
    npcs: &[NpcData],
    apprenticeships: &[ApprenticeshipRelation],
) -> SettlementKnowledgeHealth {
    let book_health = book_count_factor(books) * 0.4
                    + concept_diversity(books) * 0.3
                    + avg_condition(books) * 0.3;

    let mentorship_health = active_apprenticeship_factor(apprenticeships) * 0.5
                          + master_quality_factor(apprenticeships) * 0.5;

    let oral_health = elder_count_factor(npcs) * 0.4
                    + oral_activity_factor(npcs) * 0.6;

    SettlementKnowledgeHealth {
        overall: book_health * 0.40 + mentorship_health * 0.35 + oral_health * 0.25,
        risk_factors: identify_knowledge_risks(books, npcs, apprenticeships),  // 唯一持有者风险、书籍退化风险
    }
}
```

## 六、概念空间的代际漂移

```rust
// woworld_lang
pub fn concept_space_drift(
    concepts: &mut [CultureConceptDef],
    settlement_wisdom: &SettlementWisdomAggregate,
    external_contacts: &[CultureContactSnapshot],
    current_generation: u32,
) -> Vec<ConceptDriftEvent> {
    for concept in concepts.iter_mut() {
        let usage = settlement_wisdom.concept_usage_frequency(concept.local_id);

        // 使用频率 → 边缘化或强化
        if usage < 0.05 && concept.emergence_generation > 5.0 {
            concept.granularity *= 0.97;  // 概念边界缓慢模糊化
        } else if usage > 0.5 {
            concept.granularity = (concept.granularity * 1.01).min(1.0);
        }

        // 外部文化接触 → 概念覆盖漂移
        for contact in external_contacts.iter().filter(|c| c.intensity > 0.1) {
            if let Some(foreign) = contact.nearest_concept(concept, 0.5) {
                let influence = contact.intensity * contact.duration_years * 0.001;
                concept.coverage = concept.coverage.blend(&foreign.coverage, influence);
            }
        }
    }
}
```

## 七、完整代际故事追踪

```
第 0 代: 传奇铁匠 "老铁"
  ├─ "分段淬火" MentalModel (confidence 0.92, SelfDerived)
  ├─ WritingImpulse → 《老铁锻经》 (source_concepts 永久保存)
  ├─ 三个徒弟: DeepMentorship → confidence × 0.85
  └─ 死亡: ThoughtImprint → 铁匠铺空气中有微弱的"淬火"印记

第 10 代:
  ├─ 《老铁锻经》抄本: physical_condition 0.4, copy_generation 3
  │   → 保真度: 0.95 × 0.98³ × 0.4 = 0.36
  │   → 后世铁匠读到 → 恢复 confidence 0.36 × 0.6 = 0.22 (假说)
  ├─ 口传: "分段淬火" → 传到第 8 代时退化为 "油淬火" → 第 10 代已失传
  ├─ ThoughtImprint: 早已消散
  └─ 概念漂移: "雇佣"的 PatternCover 在 10 代间收窄——不再覆盖学徒关系
      → 出现了新概念 "行会契约"
```
